if CLIENT then
    -- Config Version Check Logic
    local function checkConfigVersionOnSpawn()
        if not Config.AllowCheckVersion then return end
        local githubURL = "https://raw.github.com/nokillnando/Starfall/main/skidmark_system/skidmark_config.txt"
        local localVersion = Version
        local maxRetries = 10

        local function doVersionCheck()
            if not http.canRequest() then
                maxRetries = maxRetries - 1
                if maxRetries <= 0 then return end
                timer.simple(0.5, doVersionCheck)
                return
            end

            http.get(githubURL,
                function(data)
                    if not (data and data ~= "") then return end

                    local versionStart = data:find('Version = "')
                    if not versionStart then return end

                    versionStart = versionStart + 11
                    local versionEnd = data:find('"', versionStart)
                    if not versionEnd or versionEnd <= versionStart then return end

                    local remoteVersion = data:sub(versionStart, versionEnd - 1)
                    local remoteParts = string.explode(".", remoteVersion)
                    local localParts = string.explode(".", localVersion)

                    local remoteMajor = tonumber(remoteParts[1]) or 0
                    local remoteMinor = tonumber(remoteParts[2]) or 0
                    local localMajor = tonumber(localParts[1]) or 0
                    local localMinor = tonumber(localParts[2]) or 0

                    if remoteMajor > localMajor or (remoteMajor == localMajor and remoteMinor > localMinor) then
                        print(Color(255, 255, 255), "Skidmark Config Update available! Local: ", Color(0, 255, 0), localVersion, Color(255, 255, 255), " | Remote: ", Color(0, 255, 0), remoteVersion)
                    else
                        print(Color(255, 255, 255), "Local Skidmark Config is up to date! Version: ", Color(0, 255, 0), localVersion)
                    end
                end,
                function(err) end
            )
        end
        timer.simple(1.5, doVersionCheck) -- Offset slightly from core check
    end

    timer.simple(1, checkConfigVersionOnSpawn)

    print("[Skidmark] System loaded. Wire FrontWheels, RearWheels, and Base to activate.")
end
