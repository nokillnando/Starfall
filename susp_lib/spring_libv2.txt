--@name susp_lib/spring_lib (v2.2 - Smart Anti-Roll Bar)
--@author Cheezus & Manus

--[[
    Version 2.2 Major Feature:
    - Added a new `AntiRollBar` class to the library.
    - The old method of using a simple spring created unwanted diagonal forces.
    - The new AntiRollBar class calculates force based ONLY on the vertical (Z-axis)
      difference between wheels and applies a pure up/down force. This correctly
      simulates a real anti-roll bar without interfering with other suspension axes.
]]

local clamp, max = math.clamp, math.max
local Spring = class("Spring")
local AntiRollBar = class("AntiRollBar") -- New class for our smart anti-roll bar
local springs = {}
local antiRollBars = {} -- Separate table to manage them
local springID = 0

-- ... (The entire Spring class from v2.1 remains unchanged here) ...
function Spring:initialize(e1, e2, lPos1, lPos2)
    if not isValid(e1) or not isValid(e2) then error("Failed to create spring: one or both entities are not valid", 2) end
    if e1 == e2 then error("Failed to create spring: cannot constrain an entity to itself", 2) end
    self.e1, self.e2 = e1, e2
    self.phys1, self.phys2 = e1:getPhysicsObject(), e2:getPhysicsObject()
    self.massCenter1 = isValid(self.phys1) and self.phys1:getMassCenter() or Vector()
    self.massCenter2 = isValid(self.phys2) and self.phys2:getMassCenter() or Vector()
    self.lPos1, self.lPos2 = lPos1 or Vector(), lPos2 or Vector()
    self.constant, self.preload, self.dampingBump, self.dampingRebound = 0, 0, 0, 0
    self.restingLength = (e1:localToWorld(self.lPos1) - e2:localToWorld(self.lPos2)):getLength()
    self.displacement, self.lastDisplacement = 0, 0
    self.id = springID
    springs[springID] = self
    springID = springID + 1
end
function Spring:setConstant(const) self.constant = max(0, const) end
function Spring:setPreload(force) self.preload = force end
function Spring:setDamping(bump, rebound)
    self.dampingBump = max(0, bump)
    self.dampingRebound = max(0, rebound or bump)
end
function Spring:setRestingLength(len) self.restingLength = max(0, len) end
function Spring:remove() springs[self.id] = nil end
-- ... (End of unchanged Spring class) ...


--- Create a new Anti-Roll Bar controller.
-- @param e1 The first entity (wheel) to connect.
-- @param e2 The second entity (wheel) to connect.
function AntiRollBar:initialize(e1, e2)
    if not isValid(e1) or not isValid(e2) then error("Failed to create anti-roll bar: one or both entities are not valid", 2) end
    self.e1 = e1
    self.e2 = e2
    self.phys1 = e1:getPhysicsObject()
    self.phys2 = e2:getPhysicsObject()
    self.massCenter1 = isValid(self.phys1) and self.phys1:getMassCenter() or Vector()
    self.massCenter2 = isValid(self.phys2) and self.phys2:getMassCenter() or Vector()
    self.constant = 0
    self.id = springID -- Can share the same ID pool
    antiRollBars[springID] = self
    springID = springID + 1
end

--- Set the anti-roll bar's stiffness constant.
-- @param const The stiffness constant. Higher = less body roll.
function AntiRollBar:setConstant(const)
    self.constant = max(0, const)
end

function AntiRollBar:remove()
    antiRollBars[self.id] = nil
end


-- Fast, optimized applyForceOffset that avoids boundary crossing
local const = (1 / 39.3701 ^ 2) * 360 / (2 * 3.1416)

local pendingForces = {}
local pendingTorques = {}
local pendingEnts = {}

local function addForceOffset(ent, force, pos, cachedMassCenter)
    local idx = ent:entIndex()
    if not pendingEnts[idx] then
        pendingEnts[idx] = ent
        pendingForces[idx] = Vector()
        pendingTorques[idx] = Vector()
    end
    pendingForces[idx] = pendingForces[idx] + force
    local off = pos - ent:localToWorld(cachedMassCenter)
    pendingTorques[idx] = pendingTorques[idx] + off:cross(force) * const
end

local dt = game.getTickInterval()

-- Main tick hook now manages both springs and anti-roll bars
hook.add("tick", "SuspensionLibTick_v2.2", function()
    -- Clear accumulation tables
    for k in pairs(pendingEnts) do pendingEnts[k] = nil end
    for k in pairs(pendingForces) do pendingForces[k] = nil end
    for k in pairs(pendingTorques) do pendingTorques[k] = nil end

    -- Spring Logic (unchanged from v2.1)
    local invDt = 1 / dt
    if dt == 0 then invDt = 0 end
    for _, spring in pairs(springs) do
        if isValid(spring.phys1) and isValid(spring.phys2) then
            local e1, e2 = spring.e1, spring.e2
            if not e1:isFrozen() or not e2:isFrozen() then
                local wPos1 = e1:localToWorld(spring.lPos1)
                local wPos2 = e2:localToWorld(spring.lPos2)
                local forceVector = wPos1 - wPos2
                local currentLength = forceVector:getLength()
                spring.displacement = currentLength - spring.restingLength
                if currentLength > 0 and not (spring.stretchOnly and spring.displacement <= 0) then
                    forceVector:normalize()
                    local springForce = -spring.constant * spring.displacement
                    local displacementRate = (spring.displacement - spring.lastDisplacement) * invDt
                    spring.lastDisplacement = spring.displacement
                    local dampingForceValue
                    if displacementRate > 0 then dampingForceValue = -spring.dampingRebound * displacementRate
                    else dampingForceValue = -spring.dampingBump * displacementRate end
                    local totalForceValue = springForce + dampingForceValue + spring.preload
                    local totalForceVector = forceVector * totalForceValue
                    if not e1:isFrozen() then addForceOffset(e1, totalForceVector, wPos1, spring.massCenter1) end
                    if not e2:isFrozen() then addForceOffset(e2, -totalForceVector, wPos2, spring.massCenter2) end
                end
            end
        else spring:remove() end
    end

    -- New Anti-Roll Bar Logic
    local WORLD_UP = Vector(0, 0, 1)
    for _, arb in pairs(antiRollBars) do
        if isValid(arb.phys1) and isValid(arb.phys2) then
            local e1, e2 = arb.e1, arb.e2
            if not e1:isFrozen() or not e2:isFrozen() then
                local pos1 = e1:getPos()
                local pos2 = e2:getPos()

                -- Calculate vertical difference only
                local verticalDifference = pos1.z - pos2.z

                -- Calculate the corrective force
                local forceMagnitude = verticalDifference * arb.constant

                -- Apply a pure vertical force to each wheel
                addForceOffset(e1, WORLD_UP * -forceMagnitude, pos1, arb.massCenter1)
                addForceOffset(e2, WORLD_UP * forceMagnitude, pos2, arb.massCenter2)
            end
        else arb:remove() end
    end

    -- Apply all accumulated forces and torques in a single batch
    for idx, ent in pairs(pendingEnts) do
        if isValid(ent) then
            local f = pendingForces[idx]
            local t = pendingTorques[idx]
            if f.x ~= 0 or f.y ~= 0 or f.z ~= 0 then ent:applyForceCenter(f) end
            if t.x ~= 0 or t.y ~= 0 or t.z ~= 0 then ent:applyTorque(t) end
        end
    end
end)


-- Return both classes so the main script can use them
return {
    Spring = Spring,
    AntiRollBar = AntiRollBar
}
