local Version = "1.14"

if CLIENT then
    
    local function initClient(config)
        -- Defaults
        local activeConfig = config or {}
        -- Helper: Determine Right Axis if not set
        if not activeConfig.RightAxis and activeConfig.ForwardAxis and activeConfig.UpAxis then
             activeConfig.RightAxis = activeConfig.ForwardAxis:cross(activeConfig.UpAxis):getNormalized()
        end
        
        local meshes = activeConfig.meshes or {} -- Not used here but kept for pattern consistency
        local baseEntity = nil
        local wheelEntities = {}
        
        -- Default Colors for prints
        local W, R, G = config.W or Color(255, 255, 255), config.R or Color(255, 0, 0), config.G or Color(0, 255, 0)
    
        -- Version Check Logic
        local function checkVersionOnSpawn(cfg)
            if not cfg.AllowCheckVersion then return end
            -- Placeholder URL - User needs to update this
            local githubURL = "https://raw.github.com/nokillnando/Starfall/main/simcadetire/tire_simulation_core.txt"
            local localVersion = Version
            local maxRetries = 10
    
            local function doVersionCheck()
                if not http.canRequest() then
                    maxRetries = maxRetries - 1
                    if maxRetries <= 0 then return end
                    timer.simple(0.5, doVersionCheck)
                    return
                end
    
                http.get(githubURL,
                    function(data)
                        if not (data and data ~= "") then
                            if cfg.AllowCheckVersion then
                                print(W, "Tire Sim Core version check failed - no data received")
                            end
                            return
                        end
    
                        local versionStart = data:find('Version = "')
                        if not versionStart then
                            if cfg.AllowCheckVersion then
                                print(W, "Tire Sim Core version check failed - version string not found")
                            end
                            return
                        end
    
                        versionStart = versionStart + 11
                        local versionEnd = data:find('"', versionStart)
                        if not versionEnd or versionEnd <= versionStart then
                            if cfg.AllowCheckVersion then
                                print(W, "Tire Sim Core version check failed - could not parse remote version")
                            end
                            return
                        end
    
                        local remoteVersion = data:sub(versionStart, versionEnd - 1)
                        local remoteParts = string.explode(".", remoteVersion)
                        local localParts = string.explode(".", localVersion)
    
                        local remoteMajor = tonumber(remoteParts[1]) or 0
                        local remoteMinor = tonumber(remoteParts[2]) or 0
                        local localMajor = tonumber(localParts[1]) or 0
                        local localMinor = tonumber(localParts[2]) or 0
    
                        if remoteMajor > localMajor or (remoteMajor == localMajor and remoteMinor > localMinor) then
                            print(W, "Check config file for updated default variables")
                            print(W, "Tire Sim Core Update available! Local: ", G, localVersion, W, " | Remote: ", G, remoteVersion)
                        else
                            print(W, "Local Tire Sim Core is up to date! Version: ", G, localVersion)
                        end
                    end,
                    function(err)
                        if cfg.AllowCheckVersion then
                            print(W, "Tire Sim Core version check failed - could not connect to remote: ", R, err)
                        end
                    end
                )
            end
            timer.simple(1, doVersionCheck)
        end
    
        checkVersionOnSpawn(activeConfig)
        
        -- HUD / Visualizer Logic
        if player() == owner() then
            local needsHud = activeConfig.ShowAxisVisualizer or activeConfig.ShowTelemetry3D or activeConfig.ShowPacejkaGraph
            if needsHud then
                enableHud(owner(), true)
    
                -- Request the base entity from the server
                timer.create("TireSim_RequestBase", 0.5, 0, function()
                    net.start("requestBase")
                    net.send()
                end)
    
                net.receive("sendBase", function()
                    baseEntity = net.readEntity()
                    
                    wheelEntities = {}
                    local count = net.readUInt(8)
                    for i=1, count do
                        local name = net.readString()
                        local ent = net.readEntity()
                        table.insert(wheelEntities, {Name = name, Entity = ent})
                    end
    
                    if baseEntity and baseEntity:isValid() then
                        timer.stop("TireSim_RequestBase")
                    end
                end)
    
                net.receive("telemetryUpdate", function()
                    local count = net.readUInt(8)
                    for i = 1, count do
                        if wheelEntities[i] then
                            wheelEntities[i].Telemetry_Mag = net.readFloat()
                            wheelEntities[i].Telemetry_Peak = net.readFloat()
                            wheelEntities[i].Telemetry_Slip = net.readFloat()
                        end
                    end
                end)

                local font = render.createFont("Trebuchet24", 24, nil, false, nil, nil, true)
                local mathFont = render.createFont("Roboto Mono", 16, 500, true)
                local smoothTelemetry = {}
                
                hook.add("drawHud", "TireSim_TelemetryVisual", function()
                    if not (baseEntity and baseEntity:isValid()) then return end
                    
                    -- Calculate high-framerate smoothing universally
                    for i, wData in ipairs(wheelEntities) do
                        if wData.Entity and wData.Entity:isValid() and wData.Telemetry_Peak and wData.Telemetry_Peak > 0 then
                            smoothTelemetry[i] = smoothTelemetry[i] or {mag=0, slip=0}
                            smoothTelemetry[i].mag = smoothTelemetry[i].mag + (math.abs(wData.Telemetry_Mag or 0) - smoothTelemetry[i].mag) * 0.4
                            smoothTelemetry[i].slip = smoothTelemetry[i].slip + (math.abs(wData.Telemetry_Slip or 0) - smoothTelemetry[i].slip) * 0.4
                        end
                    end
                    
                    -- [TELEMETRY ON-WHEEL PILLARS]
                    if activeConfig.ShowTelemetry3D then
                        for i, wData in ipairs(wheelEntities) do
                            if wData.Entity and wData.Entity:isValid() and wData.Telemetry_Peak and wData.Telemetry_Peak > 0 then
                                local ratio = math.min(1.5, smoothTelemetry[i].mag / wData.Telemetry_Peak)
                                local pos = wData.Entity:getPos()
                                local w, h = render.getGameResolution()
                                local scale = (h or 1080) / 1080
                                
                                local sPos = pos:toScreen()
                                if sPos.visible then
                                    local barWidth = 10 * scale
                                    local barMaxHeight = 40 * scale
                                    local currentHeight = barMaxHeight * math.min(1.0, ratio)
                                    
                                    local c = Color(0,255,0)
                                    if ratio > 0.85 then c = Color(255,255,0) end
                                    if ratio >= 1.0 then c = Color(255,0,0) end
                                    
                                    -- Outline
                                    render.setColor(Color(0,0,0,200))
                                    render.drawRectOutline(sPos.x - barWidth/2 - 1, sPos.y - barMaxHeight - 1, barWidth + 2, barMaxHeight + 2)
                                    -- Background
                                    render.setColor(Color(0,0,0,150))
                                    render.drawRect(sPos.x - barWidth/2, sPos.y - barMaxHeight, barWidth, barMaxHeight)
                                    -- Fill
                                    render.setColor(c)
                                    render.drawRect(sPos.x - barWidth/2, sPos.y - currentHeight, barWidth, currentHeight)
                                    
                                    -- Label
                                    render.setFont(mathFont)
                                    render.setColor(Color(255,255,255))
                                    render.drawSimpleText(sPos.x, sPos.y - barMaxHeight - 10 * scale, tostring(math.floor(ratio*100)) .. "%", 1, 1)
                                end
                            end
                        end
                    end
                    
                    -- [PACEJKA GRAPH 2D]
                    if activeConfig.ShowPacejkaGraph then
                        local w, h = render.getGameResolution()
                        if not h or h == 0 then h = 1080 end
                        local scale = h / 1080
                        
                        -- Draw the math curve for the Primary Front Tire compound
                        local activeTireData = nil
                        for axName, axProps in pairs(activeConfig.TireSetup or {}) do
                            if string.find(axName, "Front") then
                                activeTireData = activeConfig.InternalProperties[axProps.Compound]
                                break
                            end
                        end
                        if not activeTireData then activeTireData = activeConfig.InternalProperties.Sport end
                        
                        local graphX = 50 * scale
                        local graphY = 820 * scale
                        local graphWidth = 350 * scale
                        local graphHeight = 200 * scale
                        
                        render.setColor(Color(0, 0, 0, 150))
                        render.drawRect(graphX, graphY, graphWidth, graphHeight)
                        
                        -- Draw Pacejka mathematical curve
                        local B = activeTireData.Lat_B
                        local C = activeTireData.Lat_C
                        local E = activeTireData.Lat_E
                        local D = graphHeight * 0.8
                        
                        render.setColor(Color(255, 255, 255, 100))
                        local prevX, prevY
                        for x_slip = 0, 1, 0.02 do
                            local simSlip = x_slip * 0.5 -- 0 to 0.5 radians (0-30 degrees slip)
                            local fY = D * math.sin(C * math.atan(B * simSlip - E * (B * simSlip - math.atan(B * simSlip))))
                            local screenX = graphX + (x_slip * graphWidth)
                            local screenY = graphY + graphHeight - fY
                            if prevX then
                                render.drawLine(prevX, prevY, screenX, screenY)
                            end
                            prevX = screenX; prevY = screenY
                        end
                        
                        render.setFont(mathFont)
                        render.setColor(Color(255, 255, 255, 200))
                        render.drawSimpleText(graphX + 5 * scale, graphY - 25 * scale, "Tire Traction vs Slip Angle (Pacejka Curve)", 0, 0)
                        render.drawSimpleText(graphX + 5 * scale, graphY + 5 * scale, "X-Axis: Slip Angle (sliding sideways from 0° -> 30°)", 0, 0)
                        render.drawSimpleText(graphX + 5 * scale, graphY + 25 * scale, "Y-Axis: Total Grip Force provided by tire", 0, 0)
                        
                        -- Draw live dots for ALL active wheels onto the single comparative graph
                        for i, wEnt in ipairs(wheelEntities) do
                            if wEnt.Telemetry_Peak and wEnt.Telemetry_Peak > 0 then
                                local ratio = math.min(1.0, smoothTelemetry[i].mag / wEnt.Telemetry_Peak)
                                local simSlip = smoothTelemetry[i].slip or 0
                                local x_slip = math.clamp(simSlip / 0.5, 0, 1) 
                                
                                local dotX = graphX + (x_slip * graphWidth)
                                
                                -- Now physically calculate the exact Pacejka Y value for this slip!
                                local fY = D * math.sin(C * math.atan(B * simSlip - E * (B * simSlip - math.atan(B * simSlip))))
                                local dotY = graphY + graphHeight - fY
                                
                                local c = Color(0,255,0)
                                if string.find(wEnt.Name, "Rear") then c = Color(0,200,255) end
                                if ratio > 0.85 then c = Color(255,255,0) end
                                if ratio >= 1.0 then c = Color(255,0,0) end
                                
                                render.setColor(c)
                                render.drawFilledCircle(dotX, dotY, 4 * scale)
                                render.drawSimpleText(dotX + 6 * scale, dotY, wEnt.Name, 0, 1)
                            end
                        end
                    end
                end)
    
                hook.add("drawHud", "TireSim_AxisVisual", function()
                    if not activeConfig.ShowAxisVisualizer then return end
                    if not (baseEntity and baseEntity:isValid()) then return end
    
                    local function drawAxes(ent, length, labelSuffix, name)
                        if not (ent and ent:isValid()) then return end
                        
                        local pos = ent:getPos()
                        local right, up, fwd
    
                        if ent == baseEntity then
                            -- [VISUALIZER BASE]
                            -- Manual localToWorldVector using entity basis vectors
                            -- (getForward/getRight/getUp don't require physics objects)
                            -- NOTE: Source Engine's getRight() returns -Y, so we negate it.
                            local ef, er, eu = ent:getForward(), -ent:getRight(), ent:getUp()
                            local function toWorld(v)
                                return ef * v.x + er * v.y + eu * v.z
                            end
                            right = toWorld(activeConfig.RightAxis) * length
                            up = toWorld(activeConfig.UpAxis) * length
                            fwd = toWorld(activeConfig.ForwardAxis) * length
                        else
                            -- [VISUALIZER WHEEL]
                            -- Transform activeConfig.RightAxis through the base entity's orientation
                            local bef, ber, beu = baseEntity:getForward(), -baseEntity:getRight(), baseEntity:getUp()
                            local vehicleRight = bef * activeConfig.RightAxis.x + ber * activeConfig.RightAxis.y + beu * activeConfig.RightAxis.z
                            local rawRight = -ent:getRight()
                            if rawRight:dot(vehicleRight) < 0 then
                                rawRight = -rawRight
                            end
    
                            right = rawRight * length
                            up = ent:getUp() * length
                            fwd = ent:getForward() * length
                        end
    
                        local pRight = pos + right
                        local pUp = pos + up
                        local pFwd = pos + fwd
    
                        render.pushViewMatrix({ type = "3D" })
                        render.setColor(Color(255, 0, 0)); render.draw3DLine(pos, pRight)
                        render.setColor(Color(0, 255, 0)); render.draw3DLine(pos, pUp)
                        render.setColor(Color(0, 0, 255)); render.draw3DLine(pos, pFwd)
                        render.popViewMatrix()
                        
                        -- Labels
                        local sRight = pRight:toScreen()
                        local sUp = pUp:toScreen()
                        local sFwd = pFwd:toScreen()
                        local sPos = pos:toScreen()
                        
                        render.setFont(font)
                        if labelSuffix or ent == baseEntity then
                            render.setColor(Color(255,0,0)); render.drawSimpleText(sRight.x, sRight.y, "R", 1, 1)
                            render.setColor(Color(0, 255, 0)); render.drawSimpleText(sUp.x, sUp.y, "U", 1, 1)
                            render.setColor(Color(0, 0, 255)); render.drawSimpleText(sFwd.x, sFwd.y, "F", 1, 1)
                        end
                        
                        if name then
                             render.setColor(Color(255, 255, 0))
                             render.drawSimpleText(sPos.x, sPos.y - 30, name, 1, 1)
                        end
                    end
    
                    -- Draw Base Axes
                    drawAxes(baseEntity, activeConfig.VisualizerLength * 25)
    
                    -- Draw Wheel Axes
                    for _, wData in ipairs(wheelEntities) do
                        drawAxes(wData.Entity, 15, true, wData.Name)
                    end
                end)
            end
        end
    end
    
    _G.TireSimCore = {
        initClient = initClient,
        Version = Version
    }

else -- SERVER

    local function initServer(config)
        -- ============================================================================
        -- CONFIG
        -- ============================================================================
        local Config = config
        -- Helper: Determine Right Axis if not set, or verify
        -- Typically Right = Fwd x Up.
        Config.RightAxis = Config.ForwardAxis:cross(Config.UpAxis):getNormalized()

        -- ============================================================================
        -- LOGIC
        -- ============================================================================
        
        local InternalProperties = Config.InternalProperties or {}
        
        local function GetGripForces(wheelData, activeTireData, lateralSlipAngle, load)
            local Fz = load
            -- Width is applied in the main think loop.
            local peakLateralForce = Fz * ((Fz * activeTireData.Load_A1) + activeTireData.Load_A2) * activeTireData.CoF
            
            -- [ASPECT RATIO SIMULATION]
            -- 50 is treated as the neutral baseline.
            -- Lower (e.g. 30) = stiffer (higher Lat_B multiplier) = snappier turn-in.
            -- Higher (e.g. 70) = softer (lower Lat_B multiplier) = more sluggish/flexible.
            local aspectStiffnessMult = 50 / math.max(10, wheelData.AspectRatio or 50)
        
            local B, C, D, E, x
            B = activeTireData.Lat_B * aspectStiffnessMult
            C, D, E = activeTireData.Lat_C, peakLateralForce, activeTireData.Lat_E
            x = lateralSlipAngle
            local latForce = D * math.sin(C * math.atan(B * x - E * (B * x - math.atan(B * x))))
        
            return latForce
        end
        
        wire.adjustInputs(
            {"Base", "FrontWheels", "RearWheels", "ExtraFilterStatic", "ExtraFilterSemiDynamic", "Throttle", "Brake"},
            {"entity", "array", "array", "array", "array", "number", "number"}
        )
        wire.adjustOutputs(
            {"NumWheels", "WheelNames", "Loads", "GripPcts", "AbsoluteForces", "AngularDamping", "CPULoadPct", "TickSkip"},
            {"number", "array", "array", "array", "array", "array", "number", "number"}
        )
    
        local function getValidBase()
            local base = wire.ports.Base
            return (base and base:isValid()) and base or nil
        end
    
        local Wheels = {}
        local WheelHolos = {}
        local OrderedWheelNames = {}
        local NumWheels = 0
        local VehicleFilter = {}
        local CheckedEntities = {}
        local Wheelbase = 0
        local TrackWidth = 0
        local TotalRealMass = 0
    
        local thinkCounter = 0
        local skipTicks = 0
        local lastAppliedForces = {}
        local cpuLoadPct = 0
        local lastReportedMass = -1
        local lastReportedFilterCount = -1
        
        local RightAxis = Config.RightAxis 

        local function formatMass(value)
            return string.format("%.1f", value or 0)
        end

        local function addFilterEntity(ent)
            if ent and ent:isValid() and not CheckedEntities[ent] then
                CheckedEntities[ent] = true
                table.insert(VehicleFilter, ent)
                return 1
            end
            return 0
        end

        local function addConstrainedTreeToFilter(ent)
            if not (ent and ent:isValid()) then return 0 end
            local added = addFilterEntity(ent)
            local subParts = ent:getAllConstrained()
            for _, part in pairs(subParts) do
                added = added + addFilterEntity(part)
            end
            return added
        end

        local function rebuildVehicleFilter(Base)
            if not Base then return 0 end

            VehicleFilter = {}
            CheckedEntities = {}

            addConstrainedTreeToFilter(Base)

            local staticFilter = wire.ports.ExtraFilterStatic
            if staticFilter and #staticFilter > 0 then
                for _, ent in ipairs(staticFilter) do
                    addConstrainedTreeToFilter(ent)
                end
            end

            for _, name in ipairs(OrderedWheelNames) do
                local wData = Wheels[name]
                if wData and wData.Entity and wData.Entity:isValid() then
                    addFilterEntity(wData.Entity)
                end
            end

            local semiDynamicFilter = wire.ports.ExtraFilterSemiDynamic
            if semiDynamicFilter and #semiDynamicFilter > 0 then
                for _, ent in ipairs(semiDynamicFilter) do
                    addConstrainedTreeToFilter(ent)
                end
            end

            return #VehicleFilter
        end

        local function recalculateTotalRealMass(Base)
            if not Base then return false, 0, 0 end

            local totalMass = 0
            local massChecked = {}
            local function addMassTree(ent)
                if not (ent and ent:isValid()) then return end

                local tree = ent:getAllConstrained()
                table.insert(tree, ent)
                for _, part in pairs(tree) do
                    if part and part:isValid() and not massChecked[part] then
                        massChecked[part] = true
                        local phys = part:getPhysicsObject()
                        if phys and phys:isValid() then
                            totalMass = totalMass + phys:getMass()
                        end
                    end
                end
            end

            addMassTree(Base)

            local semiDynamicFilter = wire.ports.ExtraFilterSemiDynamic
            if semiDynamicFilter and #semiDynamicFilter > 0 then
                for _, ent in ipairs(semiDynamicFilter) do
                    addMassTree(ent)
                end
            end

            local previousMass = TotalRealMass
            
            -- [SPIKE FILTER] Silently ignore impossible mass spikes caused by Physgun (often adds 50,000kg)
            if totalMass > (previousMass > 0 and previousMass + 10000 or Config.TotalVehicleMass + 10000) then
                totalMass = previousMass > 0 and previousMass or Config.TotalVehicleMass
            end
            
            TotalRealMass = totalMass

            local massChanged = math.abs(TotalRealMass - lastReportedMass) > 0.1
            if massChanged then
                print(Config.W, "[TireSim] Total Physical Mass detected: ", Config.G, formatMass(TotalRealMass) .. "kg", Config.W, " (Config Virtual: ", Config.G, formatMass(Config.TotalVehicleMass) .. "kg", Config.W, ")")
                lastReportedMass = TotalRealMass
            end

            return massChanged, previousMass, TotalRealMass
        end
    
        local function updateSemiDynamicFilter()
            local semiDynamicFilter = wire.ports.ExtraFilterSemiDynamic
            if not semiDynamicFilter or #semiDynamicFilter == 0 then return end
    
            local newEnts = 0
            for _, ent in ipairs(semiDynamicFilter) do
                newEnts = newEnts + addConstrainedTreeToFilter(ent)
            end
    
            if newEnts > 0 and Config.AllowPrintFilter then
                print("[TireSim] Semi-dynamic filter found " .. newEnts .. " new entities.")
            end

            if newEnts > 0 then
                local Base = getValidBase()
                if Base then
                    recalculateTotalRealMass(Base)
                end
            end
        end
    
        local function initializeWheels()
            local Base = getValidBase()
            if not Base then return end
    
            for name, holo in pairs(WheelHolos) do if holo and holo:isValid() then holo:remove() end end
            Wheels, WheelHolos, OrderedWheelNames, NumWheels, VehicleFilter, CheckedEntities, lastAppliedForces = {}, {}, {}, 0, {}, {}, {}
    
            local minFwd, maxFwd = 9e9, -9e9
            local minRight, maxRight = 9e9, -9e9
    
            local wheelCounts = {
                Front_Left = 0, Front_Right = 0, Rear_Left = 0, Rear_Right = 0,
                Front_Center = 0, Rear_Center = 0
            }
            local axleInputNames = {"FrontWheels", "RearWheels"}
            for _, inputName in ipairs(axleInputNames) do
                local wheelArray = wire.ports[inputName]
                local axleName = inputName:gsub("Wheels", "")
                if wheelArray and #wheelArray > 0 then
                    for i, wheelEntity in ipairs(wheelArray) do
                        if wheelEntity and wheelEntity:isValid() then
                            local localPos = Base:worldToLocal(wheelEntity:getPos())
                            minFwd = math.min(minFwd, localPos.x); maxFwd = math.max(maxFwd, localPos.x)
                            minRight = math.min(minRight, localPos.y); maxRight = math.max(maxRight, localPos.y)
    
                            local sideValue = localPos:dot(RightAxis)
                            local side = "Center"; if sideValue > 0.1 then side = "Right" elseif sideValue < -0.1 then side = "Left" end
                            
                            local countKey = axleName .. "_" .. side
                            wheelCounts[countKey] = (wheelCounts[countKey] or 0) + 1
                            local nameIndex = wheelCounts[countKey]
                            local wheelName = axleName .. "_" .. side .. "_" .. nameIndex
                            local tireProps = Config.TireSetup[axleName] or {}
                            local wheelPhysObj = wheelEntity:getPhysicsObject()
                            
                            -- [SIMCADE] Set physical material to prevent Source Engine friction
                            if wheelPhysObj and wheelPhysObj:isValid() then
                                wheelPhysObj:setMaterial(Config.WheelPhysicsMaterial or "jeeptire")
                            end
                            
                            Wheels[wheelName] = {
                                Entity = wheelEntity, PhysObj = wheelPhysObj,
                                CompoundName = tireProps.Compound,
                                Width = tireProps.Width,
                                AspectRatio = tireProps.AspectRatio,
                                Load = 0,
                            }
                            lastAppliedForces[wheelName] = {cancel=Vector(0,0,0), grip=Vector(0,0,0), gripLocal=Vector(0,0,0)}
                            table.insert(OrderedWheelNames, wheelName)
                            NumWheels = NumWheels + 1
    
                            if Config.ShowWheelHolograms then
                                local holo = holograms.create(wheelEntity:getPos(), wheelEntity:getAngles(), "models/sprops/trans/wheel_d/t_wheel25.mdl", Vector(1,0.65,1))
                                holo:setParent(Base); holo:setColor(Color(200, 200, 200, 200)); holo:setMaterial("models/debug/debugwhite"); holo:suppressEngineLighting(true)
                                WheelHolos[wheelName] = holo
                            end
                        end
                    end
                end
            end
    
            if NumWheels > 1 then
                Wheelbase = math.abs(maxFwd - minFwd)
                TrackWidth = math.abs(maxRight - minRight)
            else
                Wheelbase = 0
                TrackWidth = 0
            end
            
            rebuildVehicleFilter(Base)
            recalculateTotalRealMass(Base)
            lastReportedFilterCount = #VehicleFilter
            
            print(Config.W, "[TireSim] Initialized with ", Config.G, NumWheels, Config.W, " wheels. Simcade Version " .. Version)
            wire.ports.NumWheels = NumWheels; wire.ports.WheelNames = OrderedWheelNames
        end
    
        timer.create("TireSim_SemiDynamicFilterUpdate", math.max(0.25, Config.ConstraintRefreshInterval or 2), 0, function()
            local Base = getValidBase()
            if not Base then return end

            local previousFilterCount = lastReportedFilterCount
            local filterCount = rebuildVehicleFilter(Base)
            local massChanged, previousMass, currentMass = recalculateTotalRealMass(Base)
            local filterChanged = filterCount ~= previousFilterCount

            if Config.AllowPrintFilter and filterChanged then
                print("[TireSim] Vehicle filter rebuilt with " .. filterCount .. " entities.")
            end
            if Config.AllowPrintMassRefresh and (massChanged or filterChanged) then
                local prevMassText = (previousMass and previousMass >= 0) and formatMass(previousMass) or "?"
                print(
                    Config.W,
                    "[TireSim] Refresh update: filter ",
                    Config.G,
                    tostring(previousFilterCount),
                    Config.W,
                    " -> ",
                    Config.G,
                    tostring(filterCount),
                    Config.W,
                    " | real mass ",
                    Config.G,
                    prevMassText .. "kg",
                    Config.W,
                    " -> ",
                    Config.G,
                    formatMass(currentMass) .. "kg"
                )
            end
            lastReportedFilterCount = filterCount
        end)
        
        wire.onInputsChanged = function(changed)
            if changed.Base or changed.FrontWheels or changed.RearWheels or changed.ExtraFilterStatic or changed.Throttle or changed.Brake then
                initializeWheels()
            elseif changed.ExtraFilterSemiDynamic then
                local Base = getValidBase()
                if Base then
                    rebuildVehicleFilter(Base)
                    recalculateTotalRealMass(Base)
                    lastReportedFilterCount = #VehicleFilter
                end
            end
        end
    
        initializeWheels()
    
        -- Networking
        net.receive("requestBase", function(_, ply)
            local base = getValidBase()
            if base then
                net.start("sendBase")
                net.writeEntity(base)
                net.writeUInt(#OrderedWheelNames, 8)
                for _, name in ipairs(OrderedWheelNames) do
                    local wData = Wheels[name]
                    if wData and wData.Entity and wData.Entity:isValid() then
                        net.writeString(name)
                        net.writeEntity(wData.Entity)
                    end
                end
                net.send(ply)
            end
        end)
    
        local lastVel = Vector(0,0,0)
        local thinkCounter = 0
        local lastTelemetrySend = 0
    
        hook.add("think", "TireSim_Main", function()
            local Base = getValidBase()
            if not Base or NumWheels == 0 then return end
            
            local basePhys = Base:getPhysicsObject()
            if not basePhys or not basePhys:isValid() then return end
            
            thinkCounter = thinkCounter + 1
    
            -- [MONITORING] CPU Usage
            local qUsed = quotaUsed()
            local qMax = quotaMax()
            if qMax > 0 then
                cpuLoadPct = (qUsed / qMax) * 100
            end
    
            -- [OPTIMIZATION] Fixed Update Rate
            local tickInterval = game.getTickInterval()
            local targetInterval = 1 / math.max(1, Config.UpdateRate)
            skipTicks = math.max(0, math.floor(targetInterval / tickInterval + 0.5) - 1)
            
            local shouldUpdateGrip = (thinkCounter % (skipTicks + 1) == 0)
            
            -- State Update (Pacejka & Load) - Runs at Reduced Rate
            if shouldUpdateGrip then
                local dt = tickInterval * (skipTicks + 1)
                local currentVel = basePhys:getVelocity()
                local accel = (currentVel - lastVel) / dt
                lastVel = currentVel
                local localAccel = Base:worldToLocalVector(accel)
                local effectiveVehicleMass = (TotalRealMass > 0) and TotalRealMass or Config.TotalVehicleMass
    
                local weightTransferFwd = 0
                local weightTransferRight = 0
                if Wheelbase > 0 and TrackWidth > 0 then
                    weightTransferFwd = (localAccel.x / Config.Gravity) * effectiveVehicleMass * Config.CenterOfMassHeight / Wheelbase
                    weightTransferRight = (localAccel.y / Config.Gravity) * effectiveVehicleMass * Config.CenterOfMassHeight / TrackWidth
                end
    
                local function normalizeInput(rawValue, maxValueRef)
                    local currentMax = math.max(maxValueRef, math.abs(rawValue))
                    maxValueRef = currentMax
                    return (currentMax > 0) and (math.abs(rawValue) / currentMax) or 0
                end
    
                local throttleInput, brakeInput
                if wire.ports.Throttle or wire.ports.Brake then
                    throttleInput = normalizeInput(wire.ports.Throttle or 0, Config.ThrottleMaxValue)
                    brakeInput = normalizeInput(wire.ports.Brake or 0, Config.BrakeMaxValue)
                    throttleInput = math.clamp(throttleInput, 0.0, 1.0)
                    brakeInput = math.clamp(brakeInput, 0.0, 1.0)
                else
                    local longitudinalG = localAccel.x / Config.Gravity
                    throttleInput = math.clamp(longitudinalG, 0, 1)
                    brakeInput = math.clamp(-longitudinalG, 0, 1)
                end
                
                local loads_t, grip_pcts_t, forces_t, damping_t = {}, {}, {}, {}
                local baseUp = Base:localToWorldVector(Config.UpAxis)
                local baseRight = Base:localToWorldVector(RightAxis)
                local baseAngVel = basePhys:getAngleVelocity()
                
                -- [FIX] Validate Filter
                for j=#VehicleFilter, 1, -1 do
                     local e = VehicleFilter[j]
                     if not (e and e:isValid()) then
                         table.remove(VehicleFilter, j)
                     end
                end
    
                for i, name in ipairs(OrderedWheelNames) do
                    local wheelData = Wheels[name]
                    if wheelData.Entity and wheelData.Entity:isValid() and wheelData.PhysObj and wheelData.PhysObj:isValid() then
                        local activeTireData = InternalProperties[wheelData.CompoundName] or InternalProperties.Sport
                        
                        -- Load Calc
                        local totalRatio = Config.WeightRatioFront + Config.WeightRatioRear
                        local frontRatioNormalized = Config.WeightRatioFront / totalRatio
                        local staticLoad = effectiveVehicleMass * Config.Gravity * (string.find(name, "Front") and frontRatioNormalized or (1 - frontRatioNormalized)) / (NumWheels / 2)
                        
                        local rightSign = string.find(name, "Right") and 1 or (string.find(name, "Left") and -1 or 0)
                        -- [BIKE FIX] Center wheels (rightSign=0) skip lateral weight transfer entirely.
                        -- A single centered wheel has no lateral offset to transfer weight to.
                        local isFrontWheel = string.find(name, "Front")
    
                        -- Aero
                        local contactPointVel = basePhys:getVelocityAtPoint(wheelData.Entity:getPos())
                        local trueSpeed = contactPointVel:getLength()
                        local speedMph = trueSpeed / 17.6
                        local aeroLoad = (isFrontWheel and Config.FrontAeroFactor or Config.RearAeroFactor) * effectiveVehicleMass * (speedMph / 100)^2
    
                        -- Transfer
                        local transferLoad = 0
                        if brakeInput > Config.InputDeadzone then
                            local transferForce = brakeInput * Config.BrakeTransferFactor * effectiveVehicleMass
                            transferLoad = isFrontWheel and transferForce or -transferForce
                        elseif throttleInput > Config.InputDeadzone then
                            local transferForce = throttleInput * Config.ThrottleTransferFactor * effectiveVehicleMass
                            transferLoad = isFrontWheel and -transferForce or transferForce
                        end
    
                        wheelData.Load = math.max(50, staticLoad + aeroLoad + transferLoad + (weightTransferRight * rightSign))
    
                        -- Grip Calc
                        local traceDist = Config.TireRadius + 5
                        local trace = trace.line(wheelData.Entity:getPos(), wheelData.Entity:getPos() - baseUp * traceDist, VehicleFilter)
                        
                        local latForceMag = 0
                        local slipMetric = 0
                        local gripPct = 100
                        local rollingResistance = 0
                        
                        if trace.Hit then
                            local vehicleRight = Base:localToWorldVector(RightAxis)
                            local wheelRight = wheelData.Entity:getRight()
                            if wheelRight:dot(vehicleRight) < 0 then wheelRight = -wheelRight end
                            
                            local lateralVel = contactPointVel:dot(wheelRight)
                            local surfaceGripMultiplier = activeTireData.SurfaceMultipliers[trace.MatType] or activeTireData.SurfaceMultipliers.Default
                            gripPct = math.clamp(100 * surfaceGripMultiplier, 0, 100)
                            
                            local widthFactor = wheelData.Width / 225
                            local gripBias = isFrontWheel and Config.FrontGripBias or Config.RearGripBias
                            local finalGripMultiplier = (gripPct / 100) * widthFactor * gripBias
                            
                            -- Always calculate peak force for HUD telemetry, even when parked
                            local fz = wheelData.Load
                            local peakLatRaw = fz * ((fz * activeTireData.Load_A1) + activeTireData.Load_A2) * activeTireData.CoF
                            wheelData.PeakForce = peakLatRaw * finalGripMultiplier * Config.OverallGripMultiplier
                            
                            if trueSpeed > 15 then
                                local safeSpeed = math.max(trueSpeed, 300)
                                local lateralSlipAngle = math.atan(lateralVel / safeSpeed) * math.min(trueSpeed / 300, 1.0)
                                slipMetric = lateralSlipAngle
                                
                                local latForceRaw = GetGripForces(wheelData, activeTireData, lateralSlipAngle, wheelData.Load)
                                
                                -- Target Magnitude
                                local targetMag = latForceRaw * finalGripMultiplier * Config.OverallGripMultiplier
    
                                 -- Damping
                                local prevMag = lastAppliedForces[name] and lastAppliedForces[name].mag or 0
                                local forceChange = targetMag - prevMag
                                local dampedMag = prevMag + (forceChange / math.max(1, Config.GripDampingFactor))
                                
                                -- Clamp
                                local maxOneG = effectiveVehicleMass * Config.Gravity
                                local maxForceLim = maxOneG * 1.5 
                                dampedMag = math.clamp(dampedMag, -maxForceLim, maxForceLim)
    
                                latForceMag = dampedMag
                            else
                                latForceMag = 0
                            end
                            
                            -- Rolling Resistance
                            rollingResistance = activeTireData.BaseRollingResistance
                            if trueSpeed < 2 then 
                                 wheelData.IsStatic = true
                            else
                                 wheelData.IsStatic = false
                            end
                        else
                            latForceMag = 0
                            wheelData.PeakForce = 0
                            wheelData.IsStatic = false
                            slipMetric = 0
                        end
                        
                        -- Yaw Stability Precalc
                        local yawMag = 0
                        if Config.YawStabilityFactor > 0 then
                            local yawVel = baseAngVel.z
                            if math.abs(yawVel) > 1 and trueSpeed > 0.1 then
                                local counterForce = -(contactPointVel / trueSpeed) * (math.abs(yawVel) * Config.YawStabilityFactor * 50)
                                yawMag = counterForce:dot(baseRight)
                            end
                        end
                        
                        -- Damping Update (State Checked)
                        local targetDampPos = wheelData.IsStatic and 100 or 0
                        local targetDampAng = wheelData.IsStatic and 100 or rollingResistance
                        if wheelData.LastDampAng ~= targetDampAng or wheelData.LastDampPos ~= targetDampPos then
                            wheelData.PhysObj:setDamping(targetDampPos, targetDampAng)
                            wheelData.LastDampPos = targetDampPos
                            wheelData.LastDampAng = targetDampAng
                        end
                        
                        lastAppliedForces[name] = { mag = latForceMag }
                        wheelData.LastComponents = { Mag = latForceMag, YawMag = yawMag, IsStatic = wheelData.IsStatic, Damping = rollingResistance, Slip = math.abs(slipMetric) }
                        
                        loads_t[i] = wheelData.Load
                        grip_pcts_t[i] = gripPct
                        forces_t[i] = latForceMag
                        damping_t[i] = rollingResistance
                    end
                end
                
                wire.ports.Loads, wire.ports.GripPcts, wire.ports.AbsoluteForces, wire.ports.AngularDamping = loads_t, grip_pcts_t, forces_t, damping_t
                wire.ports.CPULoadPct = cpuLoadPct
                wire.ports.TickSkip = skipTicks
                
                -- [TELEMETRY NETWORKING] 
                if Config.ShowTelemetry3D or Config.ShowPacejkaGraph then
                    -- Interval limit ensures we don't trigger the Starfall 'net burst limit' safeguard
                    local now = timer.systime()
                    if now - lastTelemetrySend >= 0.03 then -- capped at ~33Hz
                        if net.getBytesLeft and net.getBytesLeft() > 256 then
                            net.start("telemetryUpdate")
                            net.writeUInt(NumWheels, 8)
                            for _, name in ipairs(OrderedWheelNames) do
                                local wData = Wheels[name]
                                if wData and wData.Entity and wData.Entity:isValid() then
                                    net.writeFloat(wData.LastComponents and wData.LastComponents.Mag or 0)
                                    net.writeFloat(wData.PeakForce or 0)
                                    net.writeFloat(wData.LastComponents and wData.LastComponents.Slip or 0)
                                else
                                    -- Always write to keep the buffer sequence aligned
                                    net.writeFloat(0)
                                    net.writeFloat(0)
                                    net.writeFloat(0)
                                end
                            end
                            net.send(owner())
                            lastTelemetrySend = now
                        end
                    end
                end
            end 
            
            -- Force Apply Block 
            local effectiveVehicleMass = (TotalRealMass > 0) and TotalRealMass or Config.TotalVehicleMass
            local massRatio = math.min(1.0, effectiveVehicleMass / Config.TotalVehicleMass)
            massRatio = math.max(0.1, massRatio) 
            
            local vehicleRight = Base:localToWorldVector(RightAxis)
            
            for i, name in ipairs(OrderedWheelNames) do
                local wheelData = Wheels[name]
                if wheelData and wheelData.Entity and wheelData.Entity:isValid() and wheelData.PhysObj and wheelData.PhysObj:isValid() then
                    local comps = wheelData.LastComponents
                    if comps then
                        local wheelPhys = wheelData.PhysObj
                        local totalForceVec = Vector(0, 0, 0)
                        
                        -- 1. Apply Lateral Grip
                        if math.abs(comps.Mag) > 0.1 then
                             local wheelRight = wheelData.Entity:getRight()
                             if wheelRight:dot(vehicleRight) < 0 then wheelRight = -wheelRight end
                             totalForceVec = totalForceVec + (-wheelRight * comps.Mag * massRatio)
                        end
                        
                        -- 2. Yaw Stability
                        if comps.YawMag and math.abs(comps.YawMag) > 0.1 then
                             totalForceVec = totalForceVec + (vehicleRight * comps.YawMag * massRatio)
                        end
                        
                        -- Apply unified force
                        if totalForceVec.x ~= 0 or totalForceVec.y ~= 0 or totalForceVec.z ~= 0 then
                             wheelPhys:applyForceCenter(totalForceVec)
                        end
                    end
                end
            end
        end)
    end -- end initServer

    _G.TireSimCore = {
        initServer = initServer,
        Version = Version
    }
