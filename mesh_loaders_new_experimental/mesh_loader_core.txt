local function normalizeMeshes(config)
    if not config then return end
    
    -- Normalize root-level global settings first
    local defaultGlobals = getDefaultGlobalSettings()
    for k, v in pairs(defaultGlobals) do
        if config[k] == nil then
            if type(v) == "table" then
                config[k] = table.copy(v)
            else
                config[k] = v
            end
        end
    end
    
    if not config.meshes then return end
    
    local defaultCore = getDefaultConfigCore()
    local W = config.W or Color(255, 255, 255)
    local R = config.R or Color(255, 0, 0)
    local G = config.G or Color(0, 255, 0)
    
    local warnedRemoved = {}
    
    for i, meshConfig in ipairs(config.meshes) do
        -- 1. Populating missing new variables with their core defaults
        for k, v in pairs(defaultCore) do
            if meshConfig[k] == nil then
                if type(v) == "table" and k ~= "Flags" and k ~= "ColorAndAlpha" then
                    meshConfig[k] = table.copy(v)
                else
                    meshConfig[k] = v
                end
            end
        end
        
        -- 2. Checking for variables in the user's config mesh that do not exist in defaultCore
        local ignoredKeys = {
            mdl = true,
            preloadedSwaps = true,
            mat = true,
            meshData = true,
            holoMeshes = true,
            OriginalTexture = true,
            Parent = true,
            FixAngles = true,
            AngleFixTarget = true,
            TextureSwaps = true,
        }
        for k, v in pairs(meshConfig) do
            if not ignoredKeys[k] and defaultCore[k] == nil then
                if not warnedRemoved[k] then
                    warnedRemoved[k] = true
                    print(R, "Mesh Loader Warning: Variable '", W, k, R, "' in user config is deprecated/removed in Core v" .. Version)
                end
            end
        end
    end
end

    local vtfDownloads = vtfDownloads or {}

    local function hashString(str)
        local h = 0
        for i = 1, #str do
            h = (h * 31 + string.byte(str, i)) % 1000000000
        end
        return tostring(h)
    end

    local function vtfTempPath(url)
        local cleanUrl = url:match("^([^?]+)") or url
        local name = cleanUrl:match("([^/]+)%.vtf") or "unnamed"
        return "../data/sf_filedatatemp/" .. owner():getSteamID64(false) .. "/" .. name .. "_" .. hashString(cleanUrl) .. ".vtf"
    end

    local function asyncDownloadVtf(url, callback)
        if not url:lower():find("%.vtf") then
            callback(nil)
            return
        end

        local cleanUrl = url:match("^([^?]+)") or url
        local name = cleanUrl:match("([^/]+)%.vtf") or "unnamed"
        local fileName = name .. "_" .. hashString(cleanUrl) .. ".vtf"
        local tempPath = vtfTempPath(url)

        -- Stagger VTF loads to prevent Source VFS from choking on simultaneous reads
        vtfLoadDelay = (vtfLoadDelay or 0) + 1
        local myDelay = vtfLoadDelay * 0.3

        -- Double-tap: apply texture now AND re-apply after 2s in case Source cached a failed read
        local function safeCallback(path)
            if isValid(chip()) then callback(path) end
            timer.simple(2, function()
                if isValid(chip()) then callback(path) end
            end)
        end

        -- already on disk?
        if file.existsTemp(fileName) then
            timer.simple(myDelay, function()
                safeCallback(tempPath)
            end)
            return
        end

        -- somebody else already downloading it?
        if vtfDownloads[url] then
            local tries = 0
            local poll
            poll = function()
                tries = tries + 1
                if file.existsTemp(fileName) then
                    timer.simple(0.3, function()
                        safeCallback(tempPath)
                    end)
                elseif tries < 30 then
                    timer.simple(0.25, poll)
                else
                    callback(nil)
                end
            end
            timer.simple(0.1, poll)
            return
        end

        vtfDownloads[url] = true

        ------------------------------------------------------------------
        -- NEW: never break the global limit
        local function safeGet()
            local active = http.getActiveRequests()
            local max    = http.getMaximumRequests()

            if active >= max then
                -- still full  try again later
                timer.simple(0.25, safeGet)
                return
            end

            -- now we are guaranteed to have a free slot
            file.deleteTemp(fileName)

            http.get(url,
                function(data)
                    vtfDownloads[url] = nil
                    if data and #data > 0 then
                        file.writeTemp(fileName, data)
                        -- Delay before telling material to use the file
                        timer.simple(0.3, function()
                            safeCallback(tempPath)
                        end)
                    else
                        callback(nil)
                    end
                end,
                function()
                    vtfDownloads[url] = nil
                    callback(nil)
                end
            )
        end

        safeGet()
    end


    local downloadQueue = {}

    local function processDownloadQueue()
        local activeRequests = http.getActiveRequests()
        local maxRequests = http.getMaximumRequests()

        while activeRequests < maxRequests and #downloadQueue > 0 do
            local job = table.remove(downloadQueue, 1)
            asyncDownloadVtf(job.url, job.callback)
            activeRequests = activeRequests + 1
        end

        if #downloadQueue > 0 then
            timer.simple(0.1, processDownloadQueue)
        end
    end

    local vtfQueue = {}
    local vtfRunning = 0

    local function pumpVtfQueue()
        local max = http.getMaximumRequests()
        while vtfRunning < max and #vtfQueue > 0 do
            local job = table.remove(vtfQueue, 1)
            vtfRunning = vtfRunning + 1
            asyncDownloadVtf(job.url, function(path)
                vtfRunning = vtfRunning - 1
                if job.callback then job.callback(path) end
                pumpVtfQueue()               -- may start next
            end)
        end
    end

    -- drop-in replacement for every place that used queueVtfDownload
    local function queueVtfDownload(url, callback)
        table.insert(vtfQueue, { url = url, callback = callback })
        pumpVtfQueue()
    end

    local function getRestingAngle(wheelEnt, fixEnt, fixP, fixY, fixR, invertAxis)
        local restingWorldAng = fixEnt:getAngles()
        
        if type(fixP) == "number" then restingWorldAng = restingWorldAng:rotateAroundAxis(restingWorldAng:getRight(), fixP) end
        if type(fixY) == "number" then restingWorldAng = restingWorldAng:rotateAroundAxis(restingWorldAng:getUp(), fixY) end
        if type(fixR) == "number" then restingWorldAng = restingWorldAng:rotateAroundAxis(restingWorldAng:getForward(), fixR) end

        if fixP ~= nil and fixP ~= false then
            local R1 = restingWorldAng:getRight()
            local R2 = wheelEnt:getRight()
            if invertAxis then R2 = -R2 end
            local axis = R1:cross(R2)
            local len = axis:getLength()
            if len > 0.0001 then
                axis:normalize()
                local dot = math.clamp(R1:dot(R2), -1, 1)
                restingWorldAng = restingWorldAng:rotateAroundAxis(axis, math.deg(math.acos(dot)))
            elseif R1:dot(R2) < 0 then
                restingWorldAng = restingWorldAng:rotateAroundAxis(restingWorldAng:getUp(), 180)
            end
        elseif fixY ~= nil and fixY ~= false then
            local U1 = restingWorldAng:getUp()
            local U2 = wheelEnt:getUp()
            if invertAxis then U2 = -U2 end
            local axis = U1:cross(U2)
            local len = axis:getLength()
            if len > 0.0001 then
                axis:normalize()
                local dot = math.clamp(U1:dot(U2), -1, 1)
                restingWorldAng = restingWorldAng:rotateAroundAxis(axis, math.deg(math.acos(dot)))
            elseif U1:dot(U2) < 0 then
                restingWorldAng = restingWorldAng:rotateAroundAxis(restingWorldAng:getRight(), 180)
            end
        elseif fixR ~= nil and fixR ~= false then
            local F1 = restingWorldAng:getForward()
            local F2 = wheelEnt:getForward()
            if invertAxis then F2 = -F2 end
            local axis = F1:cross(F2)
            local len = axis:getLength()
            if len > 0.0001 then
                axis:normalize()
                local dot = math.clamp(F1:dot(F2), -1, 1)
                restingWorldAng = restingWorldAng:rotateAroundAxis(axis, math.deg(math.acos(dot)))
            elseif F1:dot(F2) < 0 then
                restingWorldAng = restingWorldAng:rotateAroundAxis(restingWorldAng:getUp(), 180)
            end
        else
            restingWorldAng = wheelEnt:getAngles()
        end
        
        return restingWorldAng
    end
    local refreshFixAnglesRefs
    
    local function registerFixAnglesHolo(holo)
        local cfg = holo.MeshConfig
        if not cfg or not cfg.FixAngles then return end
        
        -- Pre-cache all the config values so the render loop does zero table lookups
        local entry = {
            holo = holo,
            parentName = holo.MeshTargetParentName,
            fixTargetName = cfg.AngleFixTarget or "Base",
            offsetPos = cfg.offsetPos or activeConfig.offsetPos or Vector(0, 0, 0),
            offsetAng = cfg.offsetAngle or activeConfig.offsetAngle or Angle(0, 0, 0),
            fixP = cfg.FixAngles.pitch,
            fixY = cfg.FixAngles.yaw,
            fixR = cfg.FixAngles.roll,
            invertAxis = cfg.FixAngles.invert,
        }
        table.insert(fixAnglesHolos, entry)
        
        -- Resolve references immediately in case networking arrived before the mesh finished downloading
        refreshFixAnglesRefs()
    end
    
    refreshFixAnglesRefs = function()
        for i = #fixAnglesHolos, 1, -1 do
            local e = fixAnglesHolos[i]
            if not isValid(e.holo) then
                table.remove(fixAnglesHolos, i)
            else
                e.parent = (e.parentName and networkedEntities[e.parentName] and isValid(networkedEntities[e.parentName])) and networkedEntities[e.parentName] or base
                e.fixTarget = (e.fixTargetName ~= "Base" and networkedEntities[e.fixTargetName] and isValid(networkedEntities[e.fixTargetName])) and networkedEntities[e.fixTargetName] or base
            end
        end
    end

    local function updateHoloTransforms(holo)
        if not isValid(holo) then return end
        
        local targetParent = base
        if holo.MeshTargetParentName and networkedEntities[holo.MeshTargetParentName] and isValid(networkedEntities[holo.MeshTargetParentName]) then
            targetParent = networkedEntities[holo.MeshTargetParentName]
        end
        
        local meshOffsetPos = (holo.MeshConfig and holo.MeshConfig.offsetPos) or activeConfig.offsetPos or Vector(0, 0, 0)
        local meshOffsetAng = (holo.MeshConfig and holo.MeshConfig.offsetAngle) or activeConfig.offsetAngle or Angle(0, 0, 0)
        
        if holo.MeshConfig and holo.MeshConfig.FixAngles then
            -- Stay parented for free position interpolation
            if holo:getParent() ~= targetParent then
                holo:setParent(targetParent)
            end
            holo:setPos(targetParent:localToWorld(meshOffsetPos))
            
            local fixTargetName = holo.MeshConfig.AngleFixTarget or "Base"
            local fixTarget = base
            if fixTargetName ~= "Base" and networkedEntities[fixTargetName] and isValid(networkedEntities[fixTargetName]) then
                fixTarget = networkedEntities[fixTargetName]
            end
            
            local restingWorldAng = getRestingAngle(targetParent, fixTarget, holo.MeshConfig.FixAngles.pitch, holo.MeshConfig.FixAngles.yaw, holo.MeshConfig.FixAngles.roll, holo.MeshConfig.FixAngles.invert)
            
            local m = Matrix()
            m:setAngles(restingWorldAng)
            m:rotate(meshOffsetAng)
            
            holo:setAngles(m:getAngles())
        else
            holo:setParent(targetParent)
            holo:setPos(targetParent:localToWorld(meshOffsetPos))
            holo:setAngles(targetParent:localToWorldAngles(meshOffsetAng))
        end
    end
    
    -- Render hook (framerate): Compute and apply for frame-perfect visual smoothness
    hook.add("PreDrawOpaqueRenderables", "MeshLoader_AngleFix", function()
        for i = 1, #fixAnglesHolos do
            local e = fixAnglesHolos[i]
            if not isValid(e.holo) then
                table.remove(fixAnglesHolos, i)
            else
                local parent = e.parent
                if not isValid(parent) then parent = base end
                local fix = e.fixTarget
                if not isValid(fix) then fix = base end
                
                -- Force parenting to the stable fixTarget (Chassis/Base) instead of the spinning wheel
                if e.holo:getParent() ~= fix then
                    e.holo:setParent(fix)
                end
                
                -- Smoothly follow the wheel's interpolated position
                e.holo:setPos(parent:localToWorld(e.offsetPos))
                
                -- Calculate the stable resting angle based on chassis orientation + wheel steer/camber
                local ang = getRestingAngle(parent, fix, e.fixP, e.fixY, e.fixR, e.invertAxis)
                
                local m = Matrix()
                m:setAngles(ang)
                m:rotate(e.offsetAng)
                
                e.holo:setAngles(m:getAngles())
            end
        end
    end)

    local activeWireSwaps = {}
    
    local function applyHoloSwaps(holo)
        if not isValid(holo) or not holo.MeshConfig then return end
        
        local swapConfig = holo.MeshConfig.TextureSwaps
        if not swapConfig or not holo.ActiveSwaps then return end
        
        local defaultColor = holo.MeshConfig.ColorAndAlpha or Color(255, 255, 255, 255)
        local finalR, finalG, finalB, finalA = defaultColor.r, defaultColor.g, defaultColor.b, defaultColor.a
        local finalHighlight = holo.MeshConfig.SuppressEngineLighting or false
        local textureToSet = nil
        
        -- Priority System: Find the active swap with the highest priority
        local bestPriority = -math.huge
        local bestSwapName = nil
        local bestSwapData = nil
        
        for name, swapData in pairs(swapConfig) do
            if holo.ActiveSwaps[name] then
                local priority = (type(swapData) == "table" and swapData.priority) or 0
                if priority > bestPriority then
                    bestPriority = priority
                    bestSwapName = name
                    bestSwapData = swapData
                end
            end
        end
        
        if bestSwapName then
            local data = bestSwapData
            
            -- Handle texture
            local texture = type(data) == "table" and data.texture or (type(data) == "string" and data)
            if texture then
                textureToSet = holo.MeshConfig.preloadedSwaps and holo.MeshConfig.preloadedSwaps[bestSwapName]
            end
            
            -- Handle highlight
            if type(data) == "table" and data.highlight then
                finalHighlight = true
            end
            
            -- Handle color calculation
            local targetColor = (type(data) == "table" and data.color) or defaultColor
            finalR, finalG, finalB, finalA = targetColor.r, targetColor.g, targetColor.b, targetColor.a
            
            if type(data) == "table" and data.dynamic then
                local dynamicMin = data.dynamicMin or 0
                local dynamicMax = data.dynamicMax or 1
                local wireVal = activeWireSwaps[bestSwapName] or activeWireSwaps[data.input] or 0
                local mult = 1
                if dynamicMax ~= dynamicMin then
                    mult = math.clamp((wireVal - dynamicMin) / (dynamicMax - dynamicMin), 0, 1)
                end
                finalR = defaultColor.r + (targetColor.r - defaultColor.r) * mult
                finalG = defaultColor.g + (targetColor.g - defaultColor.g) * mult
                finalB = defaultColor.b + (targetColor.b - defaultColor.b) * mult
                finalA = defaultColor.a + (targetColor.a - defaultColor.a) * mult
            end
        end
        
        -- Apply Texture
        local mat = holo.MeshConfig.mat
        if mat and mat.setTexture then
            mat:setTexture("$basetexture", textureToSet or holo.MeshConfig.OriginalTexture or "models/debug/debugwhite")
        end
        
        -- Apply Highlight
        holo:suppressEngineLighting(finalHighlight)
        
        -- Apply Final Color
        holo:setColor(Color(finalR, finalG, finalB, finalA))
    end
    
    local function triggerHoloSwapsForConfig(v)
        for _, holo in ipairs(spawnedHolograms) do
            if isValid(holo) and holo.MeshConfig == v then
                applyHoloSwaps(holo)
            end
        end
    end

    net.receive("sendMeshSwap", function()
        local inputName = net.readString()
        local val = net.readFloat()
        
        activeWireSwaps[inputName] = val
        
        for _, holo in ipairs(spawnedHolograms) do
            if isValid(holo) and holo.MeshConfig and holo.MeshConfig.TextureSwaps then
                local configSwaps = holo.MeshConfig.TextureSwaps
                local anyMatch = false
                
                for swapKey, swapData in pairs(configSwaps) do
                    local targetInput = (type(swapData) == "table" and swapData.input) or swapKey
                    
                    if targetInput == inputName then
                        holo.ActiveSwaps = holo.ActiveSwaps or {}
                        
                        local isActive = false
                        if type(swapData) == "table" and swapData.condition ~= nil then
                            isActive = (val == swapData.condition)
                        else
                            isActive = (val > 0)
                        end
                        
                        holo.ActiveSwaps[swapKey] = isActive or nil
                        anyMatch = true
                    end
                end
                
                if anyMatch then
                    applyHoloSwaps(holo)
                end
            end
        end
    end)

    local buildThread
    local function doneBuilding(meshName, debugPrints, meshSpawnDelay, showMeshSpawnProgress, meshes, offsetPos, offsetAngle, globalScale, triangles, W, R, G, progressData, textureCount, materialCount)
        local triCount = triangles - mesh.trianglesLeft()
        local namePrefix = "Mesh Loader - " .. triCount .. " tris - " .. meshName
        setName(namePrefix)
        if debugPrints then print(W, "Finished building, using ", G, triCount, W, " tris") end
        if debugPrints then print(W, "Starting to spawn meshes with ", G, meshSpawnDelay, W, " second delay between each...") end
        
        activeConfig.offsetPos = offsetPos
        activeConfig.offsetAngle = offsetAngle

        local spawnQueue = {}
        for meshIndex, v in ipairs(meshes) do
            if v.holoMeshes and #v.holoMeshes > 0 then
                for i, holomesh in ipairs(v.holoMeshes) do
                    table.insert(spawnQueue, {
                        meshIndex = meshIndex,
                        submeshIndex = i,
                        mesh = v,
                        holomesh = holomesh
                    })
                end
            end
        end

        progressData.building = false
        progressData.spawning = true
        progressData.currentStage = "Spawning Meshes"
        progressData.spawnProgress = 0
        progressData.spawnedMeshes = 0
        progressData.totalSpawnItems = #spawnQueue

        local totalToSpawn = #spawnQueue
        local spawned = 0

        local function spawnNextMesh()
            if #spawnQueue == 0 then
                if debugPrints then print(G, "All meshes spawned successfully!") end
                progressData.spawning = false
                setName(namePrefix .. " [" .. totalToSpawn .. "M / " .. textureCount .. "T / " .. materialCount .. "Mat] - Complete")
                return
            end

            local item = table.remove(spawnQueue, 1)
            local v = item.mesh
            local holomesh = item.holomesh

            spawned = spawned + 1

            progressData.spawnedMeshes = spawned
            progressData.spawnProgress = spawned / totalToSpawn

            if showMeshSpawnProgress and debugPrints then
                print(W, "Spawning mesh ", G, item.meshIndex, W, " (submesh ", G, item.submeshIndex, W, ") - ",
                      G, spawned, W, "/", G, totalToSpawn, W, " (",
                      G, math.floor(spawned/totalToSpawn*100), W, "%)")
            end

            local assignedParentName = v.Parent
            if type(assignedParentName) ~= "string" then
                assignedParentName = nil
            end

            local holo = hologram.create(base:getPos(), base:getAngles(), "models/Combine_Helicopter/helicopter_bomb01.mdl")
            holo.MeshTargetParentName = assignedParentName
            holo.MeshConfig = v
            table.insert(spawnedHolograms, holo)
            
            -- Apply proper transforms and parenting via central function
            updateHoloTransforms(holo)
            
            -- Register into optimized render-frequency list if needed
            registerFixAnglesHolo(holo)
            
            holo:setFilterMag(v.LowPolyTexture)
            holo:setFilterMin(v.LowPolyTexture)
            holo:setMesh(holomesh)
            holo:setMeshMaterial(v.mat)
            holo:setRenderBounds(Vector(-200), Vector(200))
            holo:setScale((v.scale or Vector(1)) * globalScale)
            holo:setColor(v.ColorAndAlpha or Color(255))

            if not v.SuppressEngineLighting == true then
                holo:suppressEngineLighting(false)
            else
                holo:suppressEngineLighting(v.SuppressEngineLighting)
            end
            
            -- Immediately apply any active swaps that were networked while we were downloading/building
            if v.TextureSwaps then
                holo.ActiveSwaps = {}
                for swapName, val in pairs(activeWireSwaps) do
                    if val > 0 and v.TextureSwaps[swapName] then
                        holo.ActiveSwaps[swapName] = true
                    end
                end
                applyHoloSwaps(holo)
            end

            if #spawnQueue > 0 then
                timer.simple(meshSpawnDelay, spawnNextMesh)
            else
                if debugPrints then print(G, "All meshes spawned successfully!") end
                progressData.spawning = false
                setName(namePrefix .. " [" .. totalToSpawn .. "M / " .. textureCount .. "T / " .. materialCount .. "Mat] - Complete")
                
                -- Reconciliation timer: handles AdvDupe2 timing where wires connect AFTER the chip boots
                -- Periodically re-requests server state AND directly re-parents any mismatched holograms
                local reconcileAttempts = 0
                timer.create("meshParentReconcile", 1, 20, function()
                    if not isValid(chip()) then timer.remove("meshParentReconcile") return end
                    reconcileAttempts = reconcileAttempts + 1
                    
                    -- Re-request all states from server
                    net.start("requestBase")
                    net.send()
                    
                    -- Directly reconcile any holograms whose parent doesn't match networkedEntities
                    local allResolved = true
                    for _, holo in ipairs(spawnedHolograms) do
                        if isValid(holo) and holo.MeshTargetParentName then
                            local expectedParent = networkedEntities[holo.MeshTargetParentName]
                            if expectedParent and isValid(expectedParent) then
                                local currentParent = holo:getParent()
                                if currentParent ~= expectedParent then
                                    holo:setParent(expectedParent)
                                    local meshOffsetPos = (holo.MeshConfig and holo.MeshConfig.offsetPos) or activeConfig.offsetPos or offsetPos
                                    local meshOffsetAng = (holo.MeshConfig and holo.MeshConfig.offsetAngle) or activeConfig.offsetAngle or offsetAngle
                                    holo:setPos(expectedParent:localToWorld(meshOffsetPos))
                                    holo:setAngles(expectedParent:localToWorldAngles(meshOffsetAng))
                                end
                            else
                                allResolved = false
                            end
                        end
                    end
                    
                    if allResolved and reconcileAttempts >= 5 then
                        timer.remove("meshParentReconcile")
                    end
                end)
            end
        end
        spawnNextMesh()
    end

    local function buildMeshes(meshName, debugPrints, meshSpawnDelay, showMeshSpawnProgress, meshes, offsetPos, offsetAngle, globalScale, progressData, W, R, G)
        setName("Mesh Loader - Building meshes...")

        progressData.building = true
        progressData.currentStage = "Building Meshes"

        progressData.buildProgress = 0

                break
            end
        end

        end

        local triangles = mesh.trianglesLeft()

        buildThread = coroutine.wrap(function()

            local materialCache = {}
            local textureCountCache = {}
            local uniqueMaterials = 0

            local function trackTexture(url)
                if type(url) ~= "string" or url == "" or (url:sub(1,4) ~= "http" and not url:find("%.vtf")) then return end
                textureCountCache[url] = true
            end
            
            -- Generate a fingerprint key from all material-defining properties
            local function materialFingerprint(v)
                local parts = {
                    v.texture or "",
                    v.BumpMapLink or "",
                    v.EnvMapLink or "",
                    v.PhongExponentTexture or "",
                    v.CubeMapTexture or "",
                    tostring(v.TextureScale or 1),
                    tostring(v.EnableHalfLambert or false),
                    tostring(v.EnableBaseMapAlphaEnvMask or false),
                    tostring(v.EnableBaseMapAlphaPhongMask or false),
                    tostring(v.EnableNormalmapAlphaEnvMapMask or false),
                    tostring(v.UseEnvMapSpecularReflections or false),
                    tostring(v.UsePhong or 0),
                    tostring(v.InvertPhongMask or 0),
                    tostring(v.UseRimLight or 0),
                    tostring(v.PhongBoost or 0),
                    tostring(v.PhongExp or 0),
                    tostring(v.PhongAlbedoTint or 0),
                    tostring(v.PhongAlbedoBoost or 0),
                    tostring(v.RimLightBoost or 0),
                    tostring(v.RimLightExp or 0),
                    tostring(v.EnvMapContrast or 0),
                    tostring(v.EnvMapSaturation or 0),
                    tostring(v.EnvMapFresnel or 0),
                    tostring(v.LowPolyTexture or false),
                    v.Material or "",
                    tostring(v.UseModTexture or false),
                    v.UseIngameBumpMapp and v.BumpMapTexture or "",
                }
                -- Vectors
                if v.EnvMapShinyness then table.insert(parts, tostring(v.EnvMapShinyness)) end
                if v.PhongTint then table.insert(parts, tostring(v.PhongTint)) end
                if v.PhongFresnelRange then table.insert(parts, tostring(v.PhongFresnelRange)) end
                if v.Color2 then table.insert(parts, tostring(v.Color2)) end
                -- Flags
                if v.Flags then
                    for _, f in ipairs(v.Flags) do table.insert(parts, tostring(f)) end
                end
                
                -- Detail parameters
                table.insert(parts, v.DetailLink or "")
                table.insert(parts, tostring(v.DetailBlendFactor or 1))
                table.insert(parts, tostring(v.DetailBlendMode or 0))
                table.insert(parts, tostring(v.DetailFrame or 0))
                table.insert(parts, tostring(v.DetailScale or 4))
                if v.DetailTextureTransform then table.insert(parts, tostring(v.DetailTextureTransform)) end
                if v.DetailTint then table.insert(parts, tostring(v.DetailTint)) end
                
                return table.concat(parts, "|")
            end
            
            for k, v in ipairs(meshes) do
                progressData.builtMeshes = builtCount
                progressData.buildProgress = builtCount / #meshes

                trackTexture(v.texture)
                trackTexture(v.BumpMapLink)
                trackTexture(v.EnvMapLink)
                trackTexture(v.PhongExponentTexture)
                trackTexture(v.DetailLink)
                if v.TextureSwaps then
                    for _, sc in pairs(v.TextureSwaps) do
                        trackTexture(type(sc) == "table" and sc.texture or sc)
                    end
                end

                if v.UseModTexture then
                    -- Check material cache before creating a new one
                    local fingerprint = materialFingerprint(v)
                    local cachedEntry = materialCache[fingerprint]
                    
                    if cachedEntry and not v.TextureSwaps then
                        -- Reuse existing material (only if NO TextureSwaps are present)
                        v.mat = cachedEntry.mat
                        v.OriginalTexture = cachedEntry.originalTexture
                        v.preloadedSwaps = cachedEntry.preloadedSwaps
                        if debugPrints then print(W, "Reusing material for mesh ", G, k) end
                    else
                    local mat = material.create("VertexLitGeneric")
                    mat:setUndefined("$basetexture")
                    mat:setUndefined("$bumpmap")
                    mat:setUndefined("$envmapmask")
                    mat:setUndefined("$phongexponenttexture")
                    mat:setUndefined("$detail")

                    if v.EnableHalfLambert then
                        mat:setInt("$flags", 134217728)
                    end

                    if v.EnableBaseMapAlphaEnvMask then
                        mat:setInt("$flags", 1048576)
                    end

                    if v.EnableBaseMapAlphaPhongMask then
                        mat:setInt("$basemapalphaphongmask", 1)
                    else
                        mat:setInt("$basemapalphaphongmask", 0)
                    end

                    if v.EnableNormalmapAlphaEnvMapMask then
                        mat:setInt("$flags", 4194304)
                    end

                    local combinedFlags = 0
                    for _, flag in ipairs(v.Flags) do
                        combinedFlags = bit.bor(combinedFlags, flag)
                    end
                    mat:setInt("$flags", combinedFlags)

                    mat:setFloat("$envmapcontrast", v.EnvMapContrast)
                    mat:setFloat("$envmapsaturation", v.EnvMapSaturation)
                    mat:setVector("$envmaptint", v.EnvMapShinyness)
                    mat:setFloat("$envmapfresnel", v.EnvMapFresnel)
                            

                    mat:setInt("$phong", v.UsePhong)
                    mat:setInt("$invertphongmask", v.InvertPhongMask)
                    mat:setVector("$phongtint",v.PhongTint)
                    mat:setFloat("$phongalbedotint",v.PhongAlbedoTint)
                    mat:setFloat("$phongboost", v.PhongBoost)
                    
                    -- Detail Parameters
                    mat:setFloat("$detailblendfactor", v.DetailBlendFactor or 1)
                    mat:setInt("$detailblendmode", v.DetailBlendMode or 0)
                    mat:setInt("$detailframe", v.DetailFrame or 0)
                    mat:setFloat("$detailscale", v.DetailScale or 4)
                    mat:setMatrix("$detailtexturetransform", v.DetailTextureTransform or Matrix())
                    mat:setVector("$detailtint", v.DetailTint or Vector(1, 1, 1))
                    mat:setFloat("$phongexponent", v.PhongExp)
                    mat:setVector("$phongfresnelranges", v.PhongFresnelRange)
                    mat:setFloat("$phongalbedoboost", v.PhongAlbedoBoost)

                    mat:setInt("$rimlight", v.UseRimLight)
                    mat:setFloat("$rimlightexponent", v.RimLightExp)
                    mat:setFloat("$rimlightboost", v.RimLightBoost)

                    mat:setVector("$color2", v.Color2)

                    if v.UseEnvMapSpecularReflections == true then
                        local cubemapUpdateInterval = 2
                        local previousPosition = chip():getPos()
                        local isInitialSpawn = true
                        local function setCubeMap()
                            local currentPosition = chip():getPos()
                            if currentPosition ~= previousPosition or isInitialSpawn then
                                local Nmat = material.load(v.CubeMapTexture)
                                local newEnvMap = Nmat:getTexture("$envmap")
                                for _, mesh_val in ipairs(meshes) do
                                    if mesh_val.mat and mesh_val.mat.setTexture then
                                        mesh_val.mat:setTexture("$envmap", newEnvMap)
                                    end
                                end
                                previousPosition = currentPosition
                                isInitialSpawn = false
                            end
                        end
                        timer.simple(1, setCubeMap)
                        local function updateCubemap()
                            if not isValid(chip()) then return end
                            setCubeMap()
                            timer.simple(cubemapUpdateInterval, updateCubemap)
                        end
                        timer.simple(cubemapUpdateInterval, updateCubemap)
                    end

                                        -- ENV MAP MASK
                    if v.EnvMapLink:lower():find("%.vtf") then
                        mat:setUndefined("$envmapmask")
                        queueVtfDownload(v.EnvMapLink, function(path)
                            if path then
                                mat:setTexture("$envmapmask", path)
                            else
                                mat:setUndefined("$envmapmask")
                            end
                        end)
                    elseif v.EnvMapLink:sub(1, 4) == "http" then
                        local isImage = v.EnvMapLink:lower():find("%.png") or v.EnvMapLink:lower():find("%.jpg") or v.EnvMapLink:lower():find("%.jpeg")
                        if activeConfig.DownloadImages and isImage then
                            queueImageDownload(v.EnvMapLink, function(path)
                                if path then
                                    local dummy = material.createFromImage(path, "VertexLitGeneric")
                                    if dummy then
                                        mat:setTexture("$envmapmask", dummy:getTexture("$basetexture"))
                                    else
                                        mat:setUndefined("$envmapmask")
                                    end
                                else
                                    mat:setUndefined("$envmapmask")
                                end
                            end)
                        elseif hasPermission("material.urlcreate", v.EnvMapLink) then
                            mat:setTextureURL("$envmapmask", v.EnvMapLink, function(_, _, w, h, layout)
                                if not (w and h) then
                                    mat:setUndefined("$envmapmask")
                                    return
                                end
                                local sx, sy = 1024 / w, 1024 / h
                                layout(0, 0, w * sx, h * sy)
                            end)
                        else
                            mat:setUndefined("$envmapmask")
                        end
                    else
                        mat:setUndefined("$envmapmask")
                    end
                    
                    -- PHONG EXPONENT TEXTURE
                    if v.PhongExponentTexture:lower():find("%.vtf") then
                        mat:setUndefined("$phongexponenttexture")
                        queueVtfDownload(v.PhongExponentTexture, function(path)
                            if path then
                                mat:setTexture("$phongexponenttexture", path)
                            else
                                mat:setUndefined("$phongexponenttexture")
                            end
                        end)
                    elseif v.PhongExponentTexture:sub(1, 4) == "http" then
                        local isImage = v.PhongExponentTexture:lower():find("%.png") or v.PhongExponentTexture:lower():find("%.jpg") or v.PhongExponentTexture:lower():find("%.jpeg")
                        if activeConfig.DownloadImages and isImage then
                            queueImageDownload(v.PhongExponentTexture, function(path)
                                if path then
                                    local dummy = material.createFromImage(path, "VertexLitGeneric")
                                    if dummy then
                                        mat:setTexture("$phongexponenttexture", dummy:getTexture("$basetexture"))
                                    else
                                        mat:setUndefined("$phongexponenttexture")
                                    end
                                else
                                    mat:setUndefined("$phongexponenttexture")
                                end
                            end)
                        elseif hasPermission("material.urlcreate", v.PhongExponentTexture) then
                            mat:setTextureURL("$phongexponenttexture", v.PhongExponentTexture, function(_, _, w, h, layout)
                                if not (w and h) then
                                    mat:setUndefined("$phongexponenttexture")
                                    return
                                end
                                local sx, sy = 1024 / w, 1024 / h
                                layout(0, 0, w * sx, h * sy)
                            end)
                        else
                            mat:setUndefined("$phongexponenttexture")
                        end
                    else
                        mat:setUndefined("$phongexponenttexture")
                    end
                    
                    -- BUMPMAP
                    if v.BumpMapLink:lower():find("%.vtf") then
                        mat:setUndefined("$bumpmap")
                        queueVtfDownload(v.BumpMapLink, function(path)
                            if path then
                                mat:setTexture("$bumpmap", path)
                            else
                                mat:setUndefined("$bumpmap")
                            end
                        end)
                    elseif v.BumpMapLink:sub(1, 4) == "http" then
                        local isImage = v.BumpMapLink:lower():find("%.png") or v.BumpMapLink:lower():find("%.jpg") or v.BumpMapLink:lower():find("%.jpeg")
                        if activeConfig.DownloadImages and isImage then
                            queueImageDownload(v.BumpMapLink, function(path)
                                if path then
                                    local dummy = material.createFromImage(path, "VertexLitGeneric")
                                    if dummy then
                                        mat:setTexture("$bumpmap", dummy:getTexture("$basetexture"))
                                    else
                                        mat:setUndefined("$bumpmap")
                                    end
                                else
                                    mat:setUndefined("$bumpmap")
                                end
                            end)
                        elseif hasPermission("material.urlcreate", v.BumpMapLink) then
                            mat:setTextureURL("$bumpmap", v.BumpMapLink, function(_, _, w, h, layout)
                                if not (w and h) then
                                    mat:setUndefined("$bumpmap")
                                    return
                                end
                                local sx, sy = 1024 / w, 1024 / h
                                layout(0, 0, w * sx, h * sy)
                            end)
                        else
                            mat:setUndefined("$bumpmap")
                        end
                    elseif v.UseIngameBumpMapp then
                        local bump = material.load(v.BumpMapTexture)
                        mat:setTexture("$bumpmap", bump:getTexture("$bumpmap"))
                    else
                        mat:setUndefined("$bumpmap")
                    end
                    
                    -- DETAIL TEXTURE
                    if v.DetailLink then
                        if v.DetailLink:lower():find("%.vtf") then
                            mat:setUndefined("$detail")
                            queueVtfDownload(v.DetailLink, function(path)
                                if path then
                                    mat:setTexture("$detail", path)
                                else
                                    mat:setUndefined("$detail")
                                end
                            end)
                        elseif v.DetailLink:sub(1, 4) == "http" then
                            local isImage = v.DetailLink:lower():find("%.png") or v.DetailLink:lower():find("%.jpg") or v.DetailLink:lower():find("%.jpeg")
                            if activeConfig.DownloadImages and isImage then
                                queueImageDownload(v.DetailLink, function(path)
                                    if path then
                                        local dummy = material.createFromImage(path, "VertexLitGeneric")
                                        if dummy then
                                            mat:setTexture("$detail", dummy:getTexture("$basetexture"))
                                        else
                                            mat:setUndefined("$detail")
                                        end
                                    else
                                        mat:setUndefined("$detail")
                                    end
                                end)
                            elseif hasPermission("material.urlcreate", v.DetailLink) then
                                mat:setTextureURL("$detail", v.DetailLink, function(_, _, w, h, layout)
                                    if not (w and h) then
                                        mat:setUndefined("$detail")
                                        return
                                    end
                                    local sx, sy = 1024 / w, 1024 / h
                                    layout(0, 0, w * sx, h * sy)
                                end)
                            else
                                mat:setUndefined("$detail")
                            end
                        else
                            -- In-game texture path
                            local dMat = material.load(v.DetailLink)
                            if dMat then
                                mat:setTexture("$detail", dMat:getTexture("$basetexture"))
                            else
                                mat:setUndefined("$detail")
                            end
                        end
                    else
                        mat:setUndefined("$detail")
                    end
                    
                    -- BASE TEXTURE
                    local vtfUrl = v.texture
                    if vtfUrl:lower():find("%.vtf") then
                        mat:setTexture("$basetexture", "hunter/myplastic")
                        queueVtfDownload(vtfUrl, function(path)
                            if path then
                                mat:setTexture("$basetexture", path)
                                v.OriginalTexture = path
                                if materialCache[fingerprint] then
                                    materialCache[fingerprint].originalTexture = path
                                end
                                triggerHoloSwapsForConfig(v)
                            else
                                mat:setTexture("$basetexture", "hunter/myplastic")
                                v.OriginalTexture = "hunter/myplastic"
                                triggerHoloSwapsForConfig(v)
                            end
                        end)
                    elseif v.texture:sub(1, 4) == "http" then
                        local isImage = v.texture:lower():find("%.png") or v.texture:lower():find("%.jpg") or v.texture:lower():find("%.jpeg")
                        if activeConfig.DownloadImages and isImage then
                            mat:setTexture("$basetexture", "hunter/myplastic")
                            queueImageDownload(v.texture, function(path)
                                if path then
                                    local dummy = material.createFromImage(path, "VertexLitGeneric")
                                    if dummy then
                                        local tex = dummy:getTexture("$basetexture")
                                        mat:setTexture("$basetexture", tex)
                                        v.OriginalTexture = tex
                                        if materialCache[fingerprint] then
                                            materialCache[fingerprint].originalTexture = tex
                                        end
                                        triggerHoloSwapsForConfig(v)
                                    else
                                        mat:setTexture("$basetexture", "hunter/myplastic")
                                        v.OriginalTexture = "hunter/myplastic"
                                        triggerHoloSwapsForConfig(v)
                                    end
                                else
                                    mat:setTexture("$basetexture", "hunter/myplastic")
                                    v.OriginalTexture = "hunter/myplastic"
                                    triggerHoloSwapsForConfig(v)
                                end
                            end)
                        elseif hasPermission("material.urlcreate", v.texture) then
                            mat:setTextureURL("$basetexture", v.texture, function(_, _, w, h, layout)
                                if not (w and h) then
                                    mat:setTexture("$basetexture", "hunter/myplastic")
                                    v.OriginalTexture = "hunter/myplastic"
                                    triggerHoloSwapsForConfig(v)
                                    return
                                end
                                local sx, sy = 1024 / w, 1024 / h
                                layout(0, 0, w * sx, h * sy)
                                timer.simple(0.2, function()
                                    if isValid(chip()) and mat then
                                        v.OriginalTexture = mat:getTexture("$basetexture")
                                        if materialCache[fingerprint] then
                                            materialCache[fingerprint].originalTexture = v.OriginalTexture
                                        end
                                        triggerHoloSwapsForConfig(v)
                                    end
                                end)
                            end)
                        else
                            mat:setTexture("$basetexture", "hunter/myplastic")
                        end
                    else
                        local m = material.load(v.texture)
                        v.OriginalTexture = m:getTexture("$basetexture")
                        mat:setTexture("$basetexture", m:getTexture("$basetexture"))
                        if materialCache[fingerprint] then
                            materialCache[fingerprint].originalTexture = v.OriginalTexture
                        end
                    end
                    
                    if v.TextureSwaps then
                        v.preloadedSwaps = {}
                        for inputName, swapConfig in pairs(v.TextureSwaps) do
                            local url = type(swapConfig) == "table" and swapConfig.texture or (type(swapConfig) == "string" and swapConfig or nil)
                            
                            if url then
                                if url:lower():find("%.vtf") then
                                    queueVtfDownload(url, function(path)
                                        if path then 
                                            v.preloadedSwaps[inputName] = path 
                                            triggerHoloSwapsForConfig(v)
                                        end
                                    end)
                                elseif url:sub(1, 4) == "http" then
                                    local isImage = url:lower():find("%.png") or url:lower():find("%.jpg") or url:lower():find("%.jpeg")
                                    if activeConfig.DownloadImages and isImage then
                                        queueImageDownload(url, function(path)
                                            if path then
                                                local dummy = material.createFromImage(path, "VertexLitGeneric")
                                                if dummy then
                                                    v.preloadedSwaps[inputName] = dummy:getTexture("$basetexture") 
                                                    triggerHoloSwapsForConfig(v)
                                                end
                                            end
                                        end)
                                    else
                                        local dummy = material.create("VertexLitGeneric")
                                        dummy:setTextureURL("$basetexture", url, function(_, _, w, h, layout)
                                            if w and h then
                                                layout(0,0, w * (1024/w), h * (1024/h))
                                                timer.simple(0.2, function()
                                                    if isValid(chip()) and dummy then 
                                                        v.preloadedSwaps[inputName] = dummy:getTexture("$basetexture") 
                                                        triggerHoloSwapsForConfig(v)
                                                    end
                                                end)
                                            end
                                        end)
                                    end
                                else
                                    v.preloadedSwaps[inputName] = material.load(url):getTexture("$basetexture")
                                end
                            end
                        end
                    end


                    local textureMatrix = Matrix()
                    textureMatrix:setScale(Vector(v.TextureScale, v.TextureScale, 1))
                    mat:setMatrix("$basetexturetransform", textureMatrix)
                    v.mat = mat
                    
                    -- Store in cache ONLY if this material isn't subject to TextureSwaps
                    if not v.TextureSwaps then
                        materialCache[fingerprint] = {
                            mat = mat,
                            originalTexture = v.OriginalTexture,
                            preloadedSwaps = v.preloadedSwaps,
                        }
                    end
                    uniqueMaterials = uniqueMaterials + 1
                    end -- end of else (cache miss)
                else
                    v.mat = material.load(v.Material)
                end

                local success = false
                local trianglesBefore = mesh.trianglesLeft()
                
                local meshTable = mesh.createFromObj(v.meshData, true)
                v.holoMeshes = {}
                local meshNames = {}

                for name, _ in pairs(meshTable) do
                    table.insert(meshNames, name)
                end

                for i = 1, #meshNames do
                    local meshName = meshNames[i]
                    try(
                        function()
                            local holomesh = meshTable[meshName]
                            table.insert(v.holoMeshes, holomesh)
                            v.finalMesh = meshTable[meshName]
                            success = true
                        end,
                        function(err)
                            print(true, W, "Failed to build mesh ", G, meshName, W, ": ", R, err.message)
                        end
                    )
                end

                if success then
                    builtCount = builtCount + 1
                    if debugPrints then print(W, "Built mesh " .. k, W, ": ", G, trianglesBefore - mesh.trianglesLeft(), W, " tris") end
                end
            end
            
            local finalTextureCount = 0
            for _ in pairs(textureCountCache) do finalTextureCount = finalTextureCount + 1 end
            
            progressData.buildProgress = 1
            progressData.builtMeshes = #meshes
            return true, finalTextureCount, uniqueMaterials
        end)

        hook.add("think", "build", function()
            while chip():getQuotaAverage() < chip():getQuotaMax() / 8 do
                local ok, texCount, matCount = buildThread()
                if ok then
                    doneBuilding(meshName, debugPrints, meshSpawnDelay, showMeshSpawnProgress, meshes, offsetPos, offsetAngle, globalScale, triangles, W, R, G, progressData, texCount, matCount)
                    hook.remove("think", "build")
                    return
                end
            end
        end)
    end

    local function downloadMeshes(debugPrints, meshes, buildMeshes, meshName, meshSpawnDelay, showMeshSpawnProgress, offsetPos, offsetAngle, globalScale, downloadStatus, progressData, W, R, G)
        local downloadCount = 0
        local failCount = 0
        local alreadyDownloaded = {}
        local requestQueue = {}

        downloadStatus.total = #meshes
        downloadStatus.completed = 0
        downloadStatus.failed = {}

        progressData.downloading = true
        progressData.currentStage = "Downloading Meshes"
        progressData.totalMeshes = #meshes
        progressData.downloadedMeshes = 0
        progressData.downloadProgress = 0

        local function processQueue()
            progressData.downloadedMeshes = downloadCount
            progressData.downloadProgress = downloadCount / #meshes

            local activeRequests = http.getActiveRequests()
            local maxRequests = http.getMaximumRequests()

            while activeRequests < maxRequests and #requestQueue > 0 do
                local k, v = unpack(table.remove(requestQueue, 1))
                if debugPrints then print(W, "Downloading mesh ", G, k, W, " from URL") end

                http.get(v.mdl, function(data)
                    if data[1] == "<" or data[1] == "{" then
                        failCount = failCount + 1
                        if debugPrints then print(true, W, "Failed to download mesh ", G, k, W, ": ", R, "Invalid format") end
                    else
                        v.meshData = data
                        alreadyDownloaded[v.mdl] = k
                        downloadCount = downloadCount + 1
                        if debugPrints then print(W, "Successfully downloaded mesh ", G, k, W, ", size: ", G, math.round(#data / 1048576, 3) .. "mb") end
                    end
                    processQueue()
                end, function(err)
                    failCount = failCount + 1
                    if debugPrints then print(true, W, "HTTP error for mesh ", G, k, W, ": ", R, err) end
                    processQueue()
                end)
                activeRequests = activeRequests + 1
            end
        end

        for k, v in ipairs(meshes) do
            if not alreadyDownloaded[v.mdl] then
                table.insert(requestQueue, {k, v})
            else
                v.meshData = meshes[alreadyDownloaded[v.mdl]].meshData
                downloadCount = downloadCount + 1
                print(W, "Reused mesh data for mesh ", G, k)
            end
        end

        timer.create("downloadMeshes", 0.1, 0, function()
            if #requestQueue == 0 and http.getActiveRequests() == 0 then
                timer.remove("downloadMeshes")
                if downloadCount > 0 then
                    if debugPrints then print(W, "All available meshes downloaded: ", G, downloadCount, "/", #meshes) end
                    progressData.downloading = false
                    buildMeshes(meshName, debugPrints, meshSpawnDelay, showMeshSpawnProgress, meshes, offsetPos, offsetAngle, globalScale, progressData, W, R, G)
                else
                    if debugPrints then print(true, W, "Failed to download any meshes!") end
                    progressData.downloading = false
                end
            else
                processQueue()
            end
        end)
    end

        local function handleUpdatesAndVersionCheck(cfg)
            local githubCoreURL = "https://raw.githubusercontent.com/nokillnando/Starfall/main/mesh_loaders_new_experimental/mesh_loader_core.txt?t=" .. tostring(math.round(timer.curtime()))

            local function doCheck()
                if not http.canRequest() then
                    maxRetries = maxRetries - 1
                    if maxRetries <= 0 then
                        print(R, "Mesh Loader: Update check timed out due to HTTP limits. Loading locally.")
                        cfg._alreadyUpdated = true
                        initClient(cfg)
                        return
                    end
                    timer.simple(0.5, doCheck)
                    return
                end

                print(W, "Mesh Loader: Checking for updates...")
                http.get(githubCoreURL, function(coreData)
                    local remoteCoreVersion = nil
                    if coreData and coreData ~= "" then
                        local header = coreData:sub(1, 200)
                        local pos = header:find('Version = "', 1, true)
                        if pos then
                            local vs = pos + 11
                            local ve = header:find('"', vs, true)
                            if ve then remoteCoreVersion = header:sub(vs, ve - 1) end
                        end
                    end

                    -- Safe, retry-capable Config version checker
                    local configRetries = 10
                    local function checkConfig()
                        if not http.canRequest() then
                            configRetries = configRetries - 1
                            if configRetries <= 0 then
                                print(R, "Mesh Loader: Config update check timed out due to HTTP limits.")
                                processResults(remoteCoreVersion, nil, coreData)
                                return
                            end
                            timer.simple(0.5, checkConfig)
                            return
                        end

                        http.get(githubConfigURL, function(configData)
                            local remoteConfigVersion = nil
                            if configData and configData ~= "" then
                                local pos = configData:find('ConfigVersion = "', 1, true)
                                if pos then
                                    local vs = pos + 17
                                    local ve = configData:find('"', vs, true)
                                    if ve then remoteConfigVersion = configData:sub(vs, ve - 1) end
                                end
                            end
                            processResults(remoteCoreVersion, remoteConfigVersion, coreData, configData)
                        end, function(err)
                            print(R, "Mesh Loader Config version check HTTP error: ", err)
                            processResults(remoteCoreVersion, nil, coreData, nil)
                        end)
                    end

                    timer.simple(0.5, checkConfig)
                end, function(err)
                    print(R, "Mesh Loader Core version check HTTP error: ", err)
                    cfg._alreadyUpdated = true
                    initClient(cfg)
                end)

        -- Sanitize all URLs in meshes before any downloads
        local function sanitizeURL(url)
            if type(url) ~= "string" or url == "" then return url end
            
            -- GitHub: convert browser links to raw content links
            if url:find("github.com") and not url:find("raw.githubusercontent.com") then
                url = string.replace(url, "github.com", "raw.githubusercontent.com")
                url = string.replace(url, "/blob/", "/")
                url = string.replace(url, "/raw/", "/")
            end
            
            -- Dropbox: convert www/browser links to direct download links
            if url:find("dropbox.com") then
                url = string.replace(url, "www.dropbox.com", "dl.dropboxusercontent.com")
                url = string.replace(url, "?dl=0", "")
                url = string.replace(url, "&dl=0", "")
            end
            
            -- Google Drive: convert sharing links to direct download
            -- Handles: https://drive.google.com/file/d/FILE_ID/view?...
            if url:find("drive.google.com/file/d/") then
                local fileId = url:match("drive.google.com/file/d/([^/]+)")
                if fileId then
                    url = "https://drive.google.com/uc?export=download&id=" .. fileId
                end
            end
            -- Handles: https://drive.google.com/open?id=FILE_ID
            if url:find("drive.google.com/open%?id=") then
                local fileId = url:match("id=([^&]+)")
                if fileId then
                    url = "https://drive.google.com/uc?export=download&id=" .. fileId
                end
            end
            
            return url
        end

                        end
                    end
                end
            end

        if ply ~= owner() then return end
        net.readStream(function(data)
            if not data or data == "" then return end
            local func, err = loadstring(data, "mesh_loader_core_updated")
            if func then
                func() -- Overwrites _G.MeshLoaderCore on the server
                print("Mesh Loader: Server-side Core updated successfully!")
                if activeServerConfig then
                    _G.MeshLoaderCore.initServer(activeServerConfig)
                end
            else
                print("Mesh Loader: Server failed to compile updated core: ", err)
            end
        end)
    end)

    net.receive("streamNewConfig", function(_, ply)
        if ply ~= owner() then return end
        net.readStream(function(data)
            if not data or data == "" then return end
            
            local realInitC = _G.MeshLoaderCore.initClient
            local realInitS = _G.MeshLoaderCore.initServer
            local intercepted = nil

            _G.MeshLoaderCore.initClient = function(c) intercepted = c end
            _G.MeshLoaderCore.initServer = function(c) intercepted = c end

            local func, err = loadstring(data, "mesh_loader_config_updated")
            if func then
                func()
                if intercepted and activeServerConfig then
                    for k, v in pairs(activeServerConfig) do
                        if k ~= "ConfigVersion" and k ~= "_alreadyUpdated" and k ~= "downloadStatus" and k ~= "progressData" then
                            intercepted[k] = v
                        end
                    end
                    activeServerConfig = intercepted
                    print("Mesh Loader: Server-side Config dynamically merged successfully!")
                    
                    if _G.MeshLoaderCore.initServer then
                        _G.MeshLoaderCore.initServer(activeServerConfig)
                    end
                end
            else
                print("Mesh Loader: Server failed to compile updated config: ", err)
            end

            _G.MeshLoaderCore.initClient = realInitC
            _G.MeshLoaderCore.initServer = realInitS
        end)
    end)

    _G.MeshLoaderCore = {
        mirror = mirror,
        initServer = initServer,
        Version = Version
    }
end
hook.add("removed", "stopStreams", function()
    if net.isStreaming() then
        net.cancelStream()
    end
end)
