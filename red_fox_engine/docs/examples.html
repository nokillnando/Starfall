Example 1: Electric Powertrain

Source: addons_examples/electric_powertrain_example.txt

This addon completely replaces the engine and transmission logic to simulate an Electric Vehicle (EV). It features instant torque, regenerative braking, single-gear transmission, and custom sound synthesis.

return {
    ElectricPowertrain = {
        Enabled = false,  -- Set to true to use

        -- Configuration
        FrontMotorCount = 1,      -- Motors on front axle
        RearMotorCount = 2,       -- Motors on rear axle
        MaxMotorTorque = 900,     -- Peak torque per motor
        BaseSpeedRPM = 4000,      -- Constant-torque region ends here
        MaxMotorRPM = 16000,      -- Max motor RPM (constant-power region)
        DriveRatio = 9.5,         -- Single reduction ratio
        RegenStrength = 0.3,      -- 0..1, how much regen when holding S
        RegenBrakeBlend = 0.4,    -- 0..1, how much regen contributes to brake signal
        EfficiencyMin = 0.85,     -- Minimum efficiency at high RPM/load
        EfficiencyLossAtMaxRPM = 0.12, -- Efficiency drop at MaxMotorRPM
        DriveBias = 0.5,          -- 0 = rear bias, 1 = front bias (base split)
        VectoringStrength = 0.15, -- 0..0.3, axle bias shift on steer
        PerWheelVectoring = true, -- Enable left/right per-wheel torque scaling
        
        -- Electric Motor Sound Paths
        MotorSoundPath = "acf_engines/electric_small.wav",  -- Motor hum sound
        MotorSoundPitch = 1.0,      -- Base pitch multiplier
        MotorSoundVolume = 0.6,     -- Base volume (0-1)

        -- Auto-sync for UI (IMPORTANT: Sound config must be synced to client!)
        SyncedState = { 
            "MotorRPM", "MotorTorque", "InReverse",
            "MotorSoundPath", "MotorSoundPitch", "MotorSoundVolume",  -- Sound config for client
            "PowerDistribution"
        },

        -- Hook: Send addon flags to clients
        NetworkWrite = function(state, config, net)
            -- Suppress default engine sounds on client
            net.writeBit(1)
        end,

        -- Hook: Receive addon flags on client
        NetworkRead = function(state, config, net)
            config.SuppressDefaultSounds = (net.readBit() == 1)
        end,

        -- Wire outputs
        WireOutputs = {
            { name = "MotorRPM", source = "MotorRPM", type = "normal" },
            { name = "MotorTorque", source = "MotorTorque", type = "normal" },
            { name = "EV_InReverse", source = "InReverse", type = "normal" },
        },

        -- Electric engines: instant activation, no cranking, no sounds
        OnIgnition = function(state, config, ctx)
            local addon = config.Addons and config.Addons.ElectricPowertrain
            
            -- Electric powertrain handles its own W/S inversion and sounds
            state.SkipWSInversion = true
            config.SkipWSInversion = true

            -- Provide HUD display overrides so core HUD reflects addon specs
            do
                if addon then
                    local motorCount = (addon.FrontMotorCount or 0) + (addon.RearMotorCount or 0)
                    if motorCount < 1 then motorCount = 1 end

                    -- Peak torque per motor (Nm)
                    local peakTorquePerMotor = (addon.MaxMotorTorque or 0)
                    -- Total motor torque (all motors combined)
                    local peakTorqueTotal = peakTorquePerMotor * motorCount

                    local driveRatio = addon.DriveRatio or 1
                    -- Torque at wheels per motor and total
                    local peakTorqueAtWheelsPerMotor = peakTorquePerMotor * driveRatio
                    local peakTorqueAtWheelsTotal = peakTorqueAtWheelsPerMotor * motorCount

                    local redline = addon.MaxMotorRPM or config.Engine_Redline or 9000

                    -- Use correct conversion: hp = (Torque * RPM) / 7127  (Torque in Nm)
                    -- Compute HP from motor torque (not wheel torque) to avoid overcounting drive ratio
                    local hpPerMotor = (peakTorquePerMotor * redline) / 7127
                    local hpTotal = (peakTorqueTotal * redline) / 7127

                    config.AddonDisplay = config.AddonDisplay or {}
                    config.AddonDisplay.engineType = "Electric Motor"
                    config.AddonDisplay.cylinders = 0
                    config.AddonDisplay.displacement = 0.0
                    config.AddonDisplay.redline = redline
                    config.AddonDisplay.flywheelMass = config.Engine_FlywheelMass or 0

                    -- Provide both per-motor and total values for clarity
                    config.AddonDisplay.peakTorquePerMotor = math.floor(peakTorquePerMotor)
                    config.AddonDisplay.peakTorqueAtWheelsPerMotor = math.floor(peakTorqueAtWheelsPerMotor)
                    config.AddonDisplay.peakTorqueTotal = math.floor(peakTorqueTotal)
                    config.AddonDisplay.peakTorque = math.floor(peakTorqueTotal) -- backward-compatible (total motor torque)

                    config.AddonDisplay.hpPerMotor = math.floor(hpPerMotor)
                    config.AddonDisplay.hpTotal = math.floor(hpTotal)
                    config.AddonDisplay.hp = math.floor(hpTotal) -- backward-compatible (total hp)

                    -- Initial power distribution estimate (front %)
                    local frontCount = (addon.FrontMotorCount or 0)
                    local pd = 0
                    if motorCount > 0 then
                        pd = (frontCount / motorCount) * 100
                    end
                    config.AddonDisplay.powerDistribution = math.floor(pd)
                end
            end

            -- Electric motor: if ignition is on, motor is active (ready for torque)
            if ctx.ignition >= 1 then
                -- Force active whenever ignition is on, prevent cranking states
                
                -- Play custom electric motor sound (server side trigger)
                if not state.MotorSoundPlaying and Core.playCustomSound then
                    Core.playCustomSound(
                        "ElectricMotor", 
                        addon.MotorSoundPath or "acf_engines/electric_small.wav",
                        1, 0.05, 0.0
                    )
                    state.MotorSoundPlaying = true
                end
                
                state.ShouldPlayBell = false -- Silence default bell
                return { ignition = 1, active = true }
            else
                -- Ignition off: motor is off, no sound
                if state.MotorSoundPlaying and Core.stopCustomSound then
                    Core.stopCustomSound("ElectricMotor")
                    state.MotorSoundPlaying = false
                end
                
                state.ShouldPlayBell = false
                return { active = false, rpm = 0 }
            end
        end,

        -- Custom electric motor sound update (Client Side Hook)
        OnSoundUpdate = function(state, config, sounds)
             -- Modulate the custom sound pitch/volume based on RPM
             local snd = sounds["Custom_ElectricMotor"]
             -- Check if sound exists and is marked Active by the server
             if snd and sounds["Custom_ElectricMotor_Active"] then
                 local addon = config.Addons.ElectricPowertrain
                 
                 local throttle = state.Throttle or 0
                 local rpm = state.RPM or 0
                 local rpmRatio = math.clamp(rpm / (addon.MaxMotorRPM or 16000), 0, 1)
                 
                 -- Volume calculation: small idle hum + throttle-based volume
                 local idleVolume = 0.05
                 local throttleVolume = throttle * (addon.MotorSoundVolume or 0.6)
                 local rpmVolume = rpmRatio * 0.2
                 local finalVolume = math.clamp(idleVolume + throttleVolume + rpmVolume, 0, 1)
                 
                 -- Pitch scales with RPM
                 local dynamicPitch = 0.8 + (rpmRatio * 0.6)
                 
                 snd:setPitch(dynamicPitch * (addon.MotorSoundPitch or 1.0) * 100)
                 snd:setVolume(finalVolume)
                 
                 -- Force play if it stopped but should be active
                 if not snd:isPlaying() then
                     snd:play()
                 end
             end
        end,

        -- Silence default sounds
        OnEngineSound = function(state, config, soundData)
            -- We handle our own sounds via Custom Sound System
            if soundData.soundType == "bell" or soundData.soundType == "starter" then
                return nil -- Return nil to silence
            end
            return soundData
        end,

        -- Reverse toggle using R key (independent of brake)
        HandleCustomTransmission = function(state, config, inputs)
            local addon = config.Addons.ElectricPowertrain

            state.InReverse = state.InReverse or false
            state.LastR = state.LastR or false

            -- R key: manual reverse toggle
            if inputs.Key_R and not state.LastR then
                state.InReverse = not state.InReverse
            end
            state.LastR = inputs.Key_R
            
            -- Auto-reverse behavior: pressing opposite direction when stopped
            local speed = math.abs(state.Speed or 0)
            local Key_W = inputs.Key_W
            local Key_S = inputs.Key_S
            
            -- Track last direction key for auto-reverse
            state.LastAutoReverseW = state.LastAutoReverseW or false
            state.LastAutoReverseS = state.LastAutoReverseS or false
            
            -- If stopped and press opposite direction, switch
            if speed < 2 then  -- Basically stopped (2 km/h threshold)
                -- Forward mode: S can trigger reverse ONLY if we're not already moving forward
                if not state.InReverse and Key_S and not state.LastAutoReverseS then
                    state.InReverse = true
                -- Reverse mode: W can trigger forward ONLY if we're not moving in reverse  
                elseif state.InReverse and Key_W and not state.LastAutoReverseW then
                    state.InReverse = false
                end
            end
            
            state.LastAutoReverseW = Key_W
            state.LastAutoReverseS = Key_S

            -- Electric powertrain: always gear 1 (no gearing, direction handled by motor polarity)
            state.Gear = 1
            state.GearRatio = 1  -- Force 1:1 ratio so forward and reverse have same torque
        end,

        -- Override brake input detection to handle W/S correctly in reverse
        OnBrakeOverride = function(state, config, ctx)
            local addon = config.Addons.ElectricPowertrain
            
            -- Electric vehicle: W/S inversion disabled in core when addon enabled
            -- So brake logic is straightforward:
            -- - Forward: S = brake
            -- - Reverse: W = brake
            -- - Space = handbrake (always)
            
            if ctx.inReverse then
                -- In reverse: W brakes, S accelerates
                return { brakeInput = ctx.key_w or ctx.key_space }
            else
                -- In forward: S brakes, W accelerates
                return { brakeInput = ctx.key_s or ctx.key_space }
            end
        end,

        -- Full engine override (electric motor)
        HandleCustomEngine = function(state, config, inputs)
            local addon = config.Addons.ElectricPowertrain

            -- Electric vehicle pedal logic:
            -- Forward mode: W = accelerate, S = brake (handled by default braking)
            -- Reverse mode: S = accelerate, W = brake (handled by default braking)
            local inReverse = state.InReverse or false
            
            -- Throttle control: Shift = full power, no shift = half power
            local throttleMultiplier = inputs.Key_Shift and 1 or 0.5
            
            local throttle
            if inReverse then
                -- In reverse: S accelerates backward
                throttle = (inputs.Key_S and 1 or 0) * throttleMultiplier
            else
                -- In forward: W accelerates forward
                throttle = (inputs.Key_W and 1 or 0) * throttleMultiplier
            end
            
            local direction = inReverse and -1 or 1

            local wheelRPM = state.Flywheel_RPM or 0
            local motorRPM = math.abs(wheelRPM * addon.DriveRatio)
            motorRPM = math.min(motorRPM, addon.MaxMotorRPM)

            -- EV-style torque curve: constant torque up to BaseSpeedRPM, then constant power
            local baseRPM = math.max(1, addon.BaseSpeedRPM or 4000)
            local torquePerMotor = addon.MaxMotorTorque
            if motorRPM > baseRPM then
                -- Constant power region: torque drops proportionally to maintain constant power
                torquePerMotor = addon.MaxMotorTorque * (baseRPM / motorRPM)
            end
            local motorCount = (addon.FrontMotorCount or 0) + (addon.RearMotorCount or 0)
            if motorCount < 1 then motorCount = 1 end
            local totalTorque = torquePerMotor * motorCount
            
            -- Efficiency model: reduce torque at high RPM/load
            local efficiency = 1.0
            if (addon.EfficiencyLossAtMaxRPM or 0) > 0 then
                local rpmRatio = math.clamp(motorRPM / math.max(addon.MaxMotorRPM, 1), 0, 1)
                local effDrop = (addon.EfficiencyLossAtMaxRPM or 0.12) * rpmRatio
                efficiency = math.clamp(1 - effDrop, addon.EfficiencyMin or 0.85, 1)
            end

            -- Motor torque (always positive, direction handled by direction multiplier)
            local motorTorque = throttle * totalTorque * efficiency

            state.RPM = motorRPM
            state.Torque = motorTorque
            state.Throttle = throttle
            state.Clutch = 0
            -- Split power based on motor counts (front/rear)
            local frontShare = (addon.FrontMotorCount or 0) / motorCount
            local rearShare = (addon.RearMotorCount or 0) / motorCount

            -- Apply base drive bias and simple steering-based axle vectoring
            local baseBias = math.clamp(addon.DriveBias or 0.5, 0, 1)
            local steer = (inputs.Key_D and 1 or 0) - (inputs.Key_A and 1 or 0)
            local biasShift = (addon.VectoringStrength or 0) * steer
            local bias = math.clamp(baseBias + biasShift, 0, 1)

            if frontShare > 0 and rearShare > 0 then
                frontShare = bias
                rearShare = 1 - bias
            end
            
            -- Apply direction and drive ratio to get final power
            local drivePower = motorTorque * addon.DriveRatio * direction

            state.Flywheel_Power = drivePower
            -- Compute numeric power distribution (front % of total)
            -- Base on motor counts (front motors / total motors) so 1 front, 2 rear => 33% front
            local frontCount = (addon.FrontMotorCount or 0)
            local motorCount = math.max(1, (addon.FrontMotorCount or 0) + (addon.RearMotorCount or 0))
            local pd = math.floor((frontCount / motorCount) * 100)
            -- expose for core HUD via AddonDisplay and for auto-sync via addon state
            addon.PowerDistribution = pd
            config.AddonDisplay = config.AddonDisplay or {}
            config.AddonDisplay.powerDistribution = pd
            config.PowerDistribution = pd

            -- Per-wheel torque vectoring (simple left/right split)
            if addon.PerWheelVectoring then
                local leftScale = math.clamp(1 + biasShift, 0.5, 1.5)
                local rightScale = math.clamp(1 - biasShift, 0.5, 1.5)

                local function buildScales(wheels, chassis)
                    if not wheels or not chassis or not isValid(chassis) then return nil end
                    local scales = {}
                    local cpos = chassis:getPos()
                    local right = chassis:getRight()
                    for i, w in ipairs(wheels) do
                        if isValid(w) then
                            local side = (w:getPos() - cpos):dot(right)
                            scales[i] = (side < 0) and leftScale or rightScale
                        else
                            scales[i] = 1
                        end
                    end
                    return scales
                end

                if (addon.FrontMotorCount or 0) > 0 then
                    state.FrontWheelPowerScale = buildScales(inputs.FrontWheels, inputs.Chassis)
                end
                if (addon.RearMotorCount or 0) > 0 then
                    state.RearWheelPowerScale = buildScales(inputs.RearWheels, inputs.Chassis)
                end
            else
                state.FrontWheelPowerScale = nil
                state.RearWheelPowerScale = nil
            end

            state.Flywheel_PowerInertia = state.Flywheel_PowerInertia or 0

            addon.MotorRPM = motorRPM
            addon.MotorTorque = motorTorque
            addon.InReverse = state.InReverse

            return true
        end
    }
}

Example 2: Custom Transmission

Source: addons_examples/transmission_example.txt

This example demonstrates two different transmission types: a sequential gearbox for racing and a dual-range truck transmission.

return {
    -- ====== SEQUENTIAL TRANSMISSION (Race-like) ======
    -- Manual sequential gearbox for sporty vehicles
    -- M1 = Upshift, M2 = Downshift, S = Reverse
    SequentialTransmission = {
        Enabled = false,  -- Set to true to use
        
        -- Configuration
        AutoBlipThrottle = true,    -- Auto-blip on downshift
        BlipAmount = 0.3,           -- How much throttle for blip
        ShiftDelay = 200,           -- Milliseconds between shifts
        LastShiftTime = 0,          -- Internal: don't modify
        
        -- Hook: Override transmission logic entirely
        HandleCustomTransmission = function(state, config, inputs)
            local addon = config.Addons.SequentialTransmission
            local now = timer.curtime() * 1000  -- Convert to milliseconds
            local timeSinceShift = now - (addon.LastShiftTime or 0)
            local canShift = timeSinceShift > addon.ShiftDelay
            
            -- Reset throttle/clutch delays after shift completes
            if timeSinceShift > addon.ShiftDelay then
                state.Throttle_Delay = 0
                state.Clutch_Delay = 0
            end
            
            -- Upshift on M1
            if inputs.Key_M1 and not state.LastM1 and canShift then
                state.Gear = math.min(state.Gear + 1, #config.Gearbox_Ratios)
                addon.LastShiftTime = now
                
                state.Throttle_Delay = 1
                state.Clutch_Delay = 1
            end
            
            -- Downshift on M2
            if inputs.Key_M2 and not state.LastM2 and canShift then
                state.Gear = math.max(state.Gear - 1, -1)
                addon.LastShiftTime = now
                
                -- Auto-blip on downshift
                if addon.AutoBlipThrottle and state.Gear > 0 then
                    state.ThrottleBlip = addon.BlipAmount
                end
                
                state.Throttle_Delay = 1
                state.Clutch_Delay = 1
            end
            
            -- Reverse engagement (S key)
            if inputs.Key_S and not state.LastS and state.Gear >= 0 and state.SpeedN <= 5 then
                state.Gear = -1
                state.InReverse = true
            end
            
            -- Forward engagement (when leaving reverse)
            if inputs.Key_W and not state.LastW and state.InReverse and state.SpeedN >= -5 then
                state.Gear = 1
                state.InReverse = false
            end
            
            -- Track key presses
            state.LastM1 = inputs.Key_M1
            state.LastM2 = inputs.Key_M2
            state.LastS = inputs.Key_S
            state.LastW = inputs.Key_W

            -- Sync addon gear for outputs/sync
            addon.Gear = state.Gear
        end
    },
    
    -- ====== TRUCK TRANSMISSION (Dual-Range) ======
    -- Manual transmission with low/high range multipliers
    -- X = Low Range (2.5x torque), Z = High Range (1.0x)
    -- M1 = Upshift, M2 = Downshift, S = Reverse
    TruckTransmission = {
        Enabled = false,  -- Set to true to use
        
        -- Configuration
        LowRangeKey = "X",            -- Key to toggle low range
        HighRangeKey = "Z",           -- Key to toggle high range
        LowRangeMultiplier = 2.5,     -- 2.5x torque in low range
        HighRangeMultiplier = 1.0,    -- 1.0x (normal) in high range
        CurrentRange = "High",        -- Start in high range
        AutoBlipThrottle = true,
        BlipAmount = 0.3,
        ShiftDelay = 200,             -- Milliseconds between gear shifts
        RangeChangeDelay = 500,       -- Milliseconds between range changes
        LastShiftTime = 0,
        LastRangeChangeTime = 0,
        BaseRatios = nil,             -- Internal: stores base ratios
        InputKeyFields = { "LowRangeKey", "HighRangeKey" },
        
        -- Auto-sync range to clients
        SyncedState = { "CurrentRange", "Gear" },
        
        -- Wire outputs
        WireOutputs = {
            { name = "CurrentGear", source = "Gear", type = "normal" },
            { name = "TransmissionRange", source = "RangeValue", type = "normal" },  -- 0=High, 1=Low
        },
        
        -- Runtime
        RangeValue = 0,  -- 0 = High range, 1 = Low range (for wire output)
        Gear = 1,
        
        HandleCustomTransmission = function(state, config, inputs)
            local addon = config.Addons.TruckTransmission
            local now = timer.curtime() * 1000
            
            -- Initialize addon gear if not set
            if not addon.Gear or addon.Gear == nil then
                addon.Gear = 1
            end
            
            -- Initialize state tracking on first run
            if not state.LastM1 then state.LastM1 = false end
            if not state.LastM2 then state.LastM2 = false end
            if not state.LastS then state.LastS = false end
            if not state.LastW then state.LastW = false end
            if not state.InReverse then state.InReverse = false end
            -- Use addon gear as the persistent gear value
            if not state.Gear then state.Gear = addon.Gear end
            
            -- Store base ratios on first run
            if not addon.BaseRatios then
                addon.BaseRatios = {}
                for i, ratio in ipairs(config.Gearbox_Ratios) do
                    addon.BaseRatios[i] = ratio
                end
            end
            
            -- Range switching using core's key press tracking
            local timeSinceRangeChange = now - (addon.LastRangeChangeTime or 0)
            local canChangeRange = timeSinceRangeChange > addon.RangeChangeDelay

            local keyLowPressed = inputs.KeyPressed and inputs.KeyPressed(addon.LowRangeKey) or false
            local keyHighPressed = inputs.KeyPressed and inputs.KeyPressed(addon.HighRangeKey) or false

            if keyLowPressed and canChangeRange then
                addon.CurrentRange = "Low"
                addon.RangeValue = 1
                addon.LastRangeChangeTime = now
            end

            if keyHighPressed and canChangeRange then
                addon.CurrentRange = "High"
                addon.RangeValue = 0
                addon.LastRangeChangeTime = now
            end
            
            -- Apply range multiplier to all ratios
            local rangeMultiplier = (addon.CurrentRange == "Low") and addon.LowRangeMultiplier or addon.HighRangeMultiplier
            for i = 1, #addon.BaseRatios do
                config.Gearbox_Ratios[i] = addon.BaseRatios[i] * rangeMultiplier
            end
            
            -- Sequential shifting: M1 = Upshift, M2 = Downshift
            local timeSinceShift = now - (addon.LastShiftTime or 0)
            if timeSinceShift > addon.ShiftDelay then
                state.Throttle_Delay = 0
                state.Clutch_Delay = 0
            end
            local canShift = timeSinceShift > addon.ShiftDelay
            
            if inputs.Key_M1 and not state.LastM1 and canShift then
                state.Gear = math.min(state.Gear + 1, #addon.BaseRatios)
                addon.LastShiftTime = now
                state.Throttle_Delay = 1
                state.Clutch_Delay = 1
            end
            
            if inputs.Key_M2 and not state.LastM2 and canShift then
                state.Gear = math.max(state.Gear - 1, -1)
                addon.LastShiftTime = now
                
                -- Auto-blip throttle on downshift
                if addon.AutoBlipThrottle and state.Gear > 0 then
                    state.ThrottleBlip = addon.BlipAmount
                end
                
                state.Throttle_Delay = 1
                state.Clutch_Delay = 1
            end
            
            -- Reverse selection should NOT be bound to brake (S).
            -- Use clutch (Alt) + M2 from 1st to enter reverse, and clutch (Alt) + M1 to exit.
            local clutchHeld = inputs.Key_Alt
            if inputs.Key_M2 and not state.LastM2 and clutchHeld and state.Gear == 1 and state.SpeedN <= 5 then
                state.Gear = -1
            elseif inputs.Key_M1 and not state.LastM1 and clutchHeld and state.Gear < 0 and state.SpeedN >= -5 then
                state.Gear = 1
            end
            
            -- Keep InReverse active based on gear, but make sure reverse throttle uses W (not S)
            state.InReverse = (state.Gear < 0)
            config.ReverseWSonReverse = true
            
            -- Track key presses
            state.LastM1 = inputs.Key_M1
            state.LastM2 = inputs.Key_M2
            state.LastS = inputs.Key_S
            state.LastW = inputs.Key_W

            -- Sync addon gear for outputs/sync
            addon.Gear = state.Gear
        end
    }
}

Example 3: Supercharger

Source: addons_examples/supercharger_example.txt

This addon implements a supercharger by overriding the turbo calculation hook. Instead of exhaust-driven lag, boost is calculated directly from RPM.

return {
    Supercharger = {
        Enabled = false,  -- Set to true to use
        
        -- Supercharger Configuration
        MaxBoostPSI = 12,           -- Maximum boost pressure (PSI) at low RPM
        MinBoostPSI = 2,            -- High-RPM boost after dropoff
        BoostRPMStart = 1000,       -- RPM where boost starts building
        BoostRPMMax = 6000,         -- RPM where boost falls to MinBoostPSI
        HighRPMDropoffCurve = 1.6,  -- Higher = stronger high-RPM dropoff
        AirPressure = 14.7,         -- Atmospheric pressure (PSI)
        EfficiencyLoss = 0.15,      -- Power loss to drive supercharger (15%)
        BoostCurve = 1.2,           -- Exponential boost curve (higher = more aggressive)
        
        -- Supercharger Sound (constant whine that scales with RPM)
        SuperchargerSoundPath = "acf_extra/vehiclefx/boost/supercharger.wav",
        SuperchargerSoundVolume = 1.0,      -- Max volume
        SuperchargerIdleVolume = 0.35,      -- Audible base whine at idle
        SuperchargerSoundPitchMin = 1.0,    -- Higher base pitch
        SuperchargerSoundPitchMax = 2.5,    -- High pitched whine at redline
        
        -- Auto-sync for UI (IMPORTANT: Sound config must be synced to client!)
        SyncedState = { 
            "SuperchargerRPM", "SuperchargerBoost", "SuperchargerPSI", "SuperchargerThrottle",
            "SuperchargerSoundPath", "SuperchargerSoundVolume", "SuperchargerIdleVolume",
            "SuperchargerSoundPitchMin", "SuperchargerSoundPitchMax"
        },
        
        -- Wire outputs
        WireOutputs = {
            { name = "SuperchargerBoost", source = "SuperchargerBoost", type = "normal" },
            { name = "SuperchargerPSI", source = "SuperchargerPSI", type = "normal" },
        },
        
        -- Override turbo calculation with supercharger logic
        OnTurboCalculation = function(state, config, turboData)
            local addon = config.Addons.Supercharger
            
            -- Supercharger boost is directly tied to engine RPM (no lag)
            local rpm = turboData.engineRPM or 0
            local throttle = turboData.throttle or 0
            
            -- Calculate boost based on RPM range
            local boostRange = math.max(addon.BoostRPMMax - addon.BoostRPMStart, 1)
            local rpmRatio = math.clamp((rpm - addon.BoostRPMStart) / boostRange, 0, 1)

            -- Apply boost curve for realistic supercharger behavior (strong low-end, dropoff at high RPM)
            local dropoffCurve = addon.HighRPMDropoffCurve or 1.6
            local dropoff = rpmRatio ^ dropoffCurve
            local boostFactor = 1 - dropoff  -- High at low RPM, low at high RPM

            -- Throttle affects boost, but superchargers keep some boost even at partial throttle
            local throttleFactor = math.clamp(0.4 + (throttle * 0.6), 0, 1)

            -- Calculate PSI: high at low RPM, tapering down at high RPM
            local minBoost = addon.MinBoostPSI or 2
            local maxBoost = addon.MaxBoostPSI or 12
            local psi = (maxBoost - (maxBoost - minBoost) * dropoff) * throttleFactor
            local boost = (psi + addon.AirPressure) / addon.AirPressure
            
            -- Efficiency loss: superchargers consume engine power to run
            -- Reduce torque multiplier slightly to account for parasitic loss
            local efficiencyFactor = 1 - (addon.EfficiencyLoss * boostFactor)
            local torqueMultiplier = turboData.torqueMultiplier * (boost / 1.25) * efficiencyFactor
            
            -- Store for sync/wire
            addon.SuperchargerRPM = rpm  -- Supercharger spins at engine speed (simplified)
            addon.SuperchargerPSI = psi
            addon.SuperchargerBoost = boost
            addon.SuperchargerThrottle = throttle
            
            return {
                turboPSI = psi,           -- Override turbo PSI
                turboBoost = boost,       -- Override turbo boost
                turboRPM = rpm,           -- Supercharger RPM = engine RPM
                torqueMultiplier = torqueMultiplier, -- Apply boost with efficiency loss
                disableBOV = true         -- Superchargers don't use a BOV
            }
        end,
        
        -- Hook: Send supercharger sound data to clients
        NetworkWrite = function(state, config, net)
            local addon = config.Addons.Supercharger
            net.writeString(addon.SuperchargerSoundPath or "")
            net.writeFloat(addon.SuperchargerSoundVolume or 1.0)
            net.writeFloat(addon.SuperchargerIdleVolume or 0.35)
            net.writeFloat(addon.SuperchargerSoundPitchMin or 1.0)
            net.writeFloat(addon.SuperchargerSoundPitchMax or 2.5)
        end,
        
        -- Hook: Receive supercharger sound data on client and create sound object
        NetworkRead = function(state, config, net)
            local addon = config.Addons.Supercharger
            addon.SuperchargerSoundPath = net.readString()
            addon.SuperchargerSoundVolume = net.readFloat()
            addon.SuperchargerIdleVolume = net.readFloat()
            addon.SuperchargerSoundPitchMin = net.readFloat()
            addon.SuperchargerSoundPitchMax = net.readFloat()
            
            -- Create or replace sound object on client side
            if addon.SuperchargerSoundPath and addon.SuperchargerSoundPath ~= "" then
                if addon._soundObject and addon._soundPath ~= addon.SuperchargerSoundPath then
                    addon._soundObject:stop()
                    addon._soundObject = nil
                end

                if not addon._soundObject then
                    -- Get engine entity - the chip's parent should be the engine
                    local engineEnt = chip():getParent()
                    if engineEnt and engineEnt:isValid() then
                        addon._soundObject = sound.create(engineEnt, addon.SuperchargerSoundPath, true)
                        addon._soundPath = addon.SuperchargerSoundPath
                        if addon._soundObject then
                            addon._soundObject:play()
                            addon._soundObject:setVolume(0)
                        end
                    end
                end
            end

            -- Create a client think hook once to drive pitch/volume updates
            if not addon._soundThinkHook then
                addon._soundThinkHook = true
                hook.add("think", "SuperchargerSound", function()
                    if not addon._soundObject then return end

                    local rpm = addon.SuperchargerRPM or 0
                    local throttle = addon.SuperchargerThrottle or 0
                    local redline = config.Engine_Redline or 9000
                    local rpmRatio = math.clamp(rpm / redline, 0, 1)

                    local pitchMin = addon.SuperchargerSoundPitchMin or 1.0
                    local pitchMax = addon.SuperchargerSoundPitchMax or 2.5
                    local pitch = pitchMin + (rpmRatio * (pitchMax - pitchMin))

                    local idleVolume = addon.SuperchargerIdleVolume or 0.35
                    local throttleVolume = throttle * (addon.SuperchargerSoundVolume or 1.0)
                    local rpmVolume = rpmRatio * 0.4
                    local volume = math.clamp(idleVolume + throttleVolume + rpmVolume, 0, 1)

                    volume = volume * (config.Engine_SoundOverallDiv or 1)

                    if rpm > 100 then
                        addon._soundObject:setPitch(math.max(0.1, pitch))
                        addon._soundObject:setVolume(math.max(0, volume))
                    else
                        addon._soundObject:setVolume(0)
                    end
                end)
            end
        end,
    }
}
