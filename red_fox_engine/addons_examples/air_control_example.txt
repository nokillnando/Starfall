-- Gyroscopic Air Control Addon Example
-- Allows player to control chassis rotation while airborne
-- Like a motorcycle or skateboard - tilt forward/back, rotate left/right
--
-- Usage: Enable in red_fox_engine_config.txt:
--   Addons.AirControl.Enabled = true

return {
    AirControl = {
        Enabled = false,  -- Set to true to use
        
        -- Configuration
        PitchControl = 5000,            -- Strength of forward/backward tilt (W/S)
        YawControl = 3000,              -- Strength of left/right rotation (A/D)
        RollControl = 2000,             -- Strength of barrel roll (Shift/Alt)
        
        -- Control inputs (set these based on your input system)
        UseWASD = true,                 -- Use WASD for control (vs manual wire inputs)
        PitchSensitivity = 1.0,         -- Multiplier for pitch response
        YawSensitivity = 1.0,           -- Multiplier for yaw response
        RollSensitivity = 1.0,          -- Multiplier for roll response
        
        -- Air detection
        GroundCheckDistance = 50,       -- Distance to check for ground (units)
        MinAirTime = 0.1,               -- Minimum airtime before control activates (seconds)
        
        -- Damping
        AngularDamping = 0.95,          -- Reduce angular velocity over time (0-1, 1=no damping)
        AutoStabilize = false,          -- Automatically stabilize to horizontal
        StabilizeStrength = 1000,       -- Strength of auto-stabilization
        
        -- Wire inputs for external control (if UseWASD = false)
        WireInputs = {
            { name = "PitchInput", target = "ExternalPitchInput", type = "normal" },
            { name = "YawInput", target = "ExternalYawInput", type = "normal" },
            { name = "RollInput", target = "ExternalRollInput", type = "normal" },
        },
        
        -- Wire outputs for monitoring
        WireOutputs = {
            { name = "IsAirborne", source = "IsAirborne", type = "normal" },
            { name = "AirTime", source = "AirTime", type = "normal" },
            { name = "GroundDist", source = "GroundDistance", type = "normal" },
        },
        
        -- Status tracking
        IsAirborne = false,
        AirTime = 0,
        GroundDistance = 0,
        LastGroundCheckTime = 0,
        ExternalPitchInput = 0,
        ExternalYawInput = 0,
        ExternalRollInput = 0,
        
        -- Hook: Apply air control forces
        OnPhysicsUpdate = function(state, config, physicsData)
            local addon = config.Addons.AirControl
            local ChassisPhys = physicsData.ChassisPhys
            local Chassis = physicsData.Chassis
            
            -- Ground check using trace
            local chassisPos = Chassis:getPos()
            local traceDown = trace.trace(chassisPos, chassisPos - Vector(0, 0, addon.GroundCheckDistance), Chassis)
            local distToGround = traceDown.HitPos and (chassisPos - traceDown.HitPos):getLength() or addon.GroundCheckDistance
            
            addon.GroundDistance = distToGround
            addon.IsAirborne = distToGround > 5  -- Consider airborne if >5 units off ground
            
            -- Track airtime
            if addon.IsAirborne then
                addon.AirTime = addon.AirTime + (1/66)  -- Approximate delta time
            else
                addon.AirTime = 0
            end
            
            -- Only apply control if airborne and past minimum air time
            if not addon.IsAirborne or addon.AirTime < addon.MinAirTime then
                return
            end
            
            -- Get control inputs
            local pitchInput, yawInput, rollInput = 0, 0, 0
            
            if addon.UseWASD then
                -- Read from state (WASD keys)
                local keyW = state.KeyInputs and state.KeyInputs.W or 0
                local keyS = state.KeyInputs and state.KeyInputs.S or 0
                local keyA = state.KeyInputs and state.KeyInputs.A or 0
                local keyD = state.KeyInputs and state.KeyInputs.D or 0
                local keyShift = state.KeyInputs and state.KeyInputs.Shift or 0
                local keyAlt = state.KeyInputs and state.KeyInputs.Alt or 0
                
                pitchInput = (keyW - keyS)  -- W = pitch down (nose down), S = pitch up
                yawInput = (keyD - keyA)    -- D = yaw right, A = yaw left
                rollInput = (keyShift - keyAlt)  -- Shift = roll right, Alt = roll left
            else
                -- Use wire inputs
                pitchInput = addon.ExternalPitchInput or 0
                yawInput = addon.ExternalYawInput or 0
                rollInput = addon.ExternalRollInput or 0
            end
            
            -- Apply sensitivity
            pitchInput = pitchInput * addon.PitchSensitivity
            yawInput = yawInput * addon.YawSensitivity
            rollInput = rollInput * addon.RollSensitivity
            
            -- Calculate torques in local space
            local pitchTorque = Chassis:getRight() * pitchInput * addon.PitchControl
            local yawTorque = Chassis:getUp() * yawInput * addon.YawControl
            local rollTorque = Chassis:getForward() * rollInput * addon.RollControl
            
            -- Combine torques
            local totalTorque = pitchTorque + yawTorque + rollTorque
            totalTorque = totalTorque * physicsData.tickMult
            
            -- Apply control torque
            if totalTorque:getLength() > 0.1 then
                ChassisPhys:applyTorque(totalTorque)
            end
            
            -- Angular damping (slow down rotation)
            if addon.AngularDamping < 1.0 then
                local angVel = ChassisPhys:getAngleVelocity()
                ChassisPhys:setAngleVelocity(angVel * addon.AngularDamping)
            end
            
            -- Auto-stabilize to horizontal
            if addon.AutoStabilize then
                local chassisUp = Chassis:getUp()
                local worldUp = Vector(0, 0, 1)
                
                -- Calculate angle difference from horizontal
                local cross = chassisUp:cross(worldUp)
                local stabilizeTorque = cross * addon.StabilizeStrength * physicsData.tickMult
                
                ChassisPhys:applyTorque(stabilizeTorque)
            end
        end,
    }
}
